// ============================================================
// CORVID 2.0 | managers/macro/macro-manager
// Stores and executes macro scripts.
// Macros live in /Config/Macros/ on the pivot session.
// The 'init' macro runs automatically on startup.
// Trigger any macro mid-session with @macroname.
// ============================================================

MacroManager = {}
MacroManager.mediator = null

MacroManager.getMacroFolder = function()
    sessionManager = self.mediator.getExtension("sessionManager")
    handler        = sessionManager.pivotSession.handler

    user   = handler.user
    cfgPath = "/home/guest/Config"
    if user == "root" then
        cfgPath = "/root/Config"
    else if user != "guest" then
        cfgPath = "/home/" + user + "/Config"
    end if

    folder = handler.getFile(cfgPath)
    if folder isa Error then
        return folder
    end if

    macroFolder = handler.getFile(cfgPath + "/Macros")
    if macroFolder isa Error then
        result = handler.createFolder(cfgPath + "/Macros")
        if result isa Error then
            return result
        end if
        macroFolder = handler.getFile(cfgPath + "/Macros")
        if macroFolder isa Error then
            return macroFolder
        end if
    end if

    return macroFolder
end function

MacroManager.getMacro = function(name)
    folder = self.getMacroFolder()
    if folder isa Error then
        return folder
    end if
    macro = folder.child(name)
    if macro isa Error then
        return Error.New("Macro not found: " + name, "NOT_FOUND")
    end if
    content = macro.content()
    if content isa Error then
        return content
    end if
    return content
end function

MacroManager.runMacro = function(name)
    commandManager = self.mediator.getExtension("commandManager")
    content = self.getMacro(name)
    if content isa Error then
        return content
    end if
    lines = content.split(char(10))
    for line in lines
        trimmed = line.trim
        if trimmed.len == 0 then
            continue
        end if
        if startsWith(trimmed, "//") then
            continue
        end if
        commandManager.execute(trimmed)
        commandManager.run()
    end for
end function

MacroManager.init = function()
    // Run init macro if it exists  -  silently skip if not
    macro = self.getMacro("init")
    if macro isa Error then
        return
    end if
    self.runMacro("init")
end function

MacroManager.listMacros = function()
    folder = self.getMacroFolder()
    if folder isa Error then
        return []
    end if
    kids = folder.children()
    if kids isa Error then
        return []
    end if
    names = []
    for kid in kids
        names.push(kid.name())
    end for
    return names
end function

MacroManager.saveMacro = function(name, content)
    folder = self.getMacroFolder()
    if folder isa Error then
        return folder
    end if
    sessionManager = self.mediator.getExtension("sessionManager")
    handler        = sessionManager.pivotSession.handler

    existing = folder.child(name)
    if existing isa Error then
        result = handler.createFile(folder.path() + "/" + name)
        if result isa Error then
            return result
        end if
    end if

    file = handler.getFile(folder.path() + "/" + name)
    if file isa Error then
        return file
    end if
    return file.setContent(content)
end function

MacroManager.deleteMacro = function(name)
    folder = self.getMacroFolder()
    if folder isa Error then
        return folder
    end if
    macro = folder.child(name)
    if macro isa Error then
        return Error.New("Macro not found: " + name, "NOT_FOUND")
    end if
    return macro.delete()
end function

MacroManager.setMediator = function(mediator)
    self.mediator = mediator
end function

MacroManager.New = function()
    result = new MacroManager
    return result
end function
