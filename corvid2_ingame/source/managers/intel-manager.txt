// ============================================================
// CORVID 2.0 | managers/intel/intel-manager
// Harvest orchestration. Reads passwd, Mail.txt, Bank.txt
// from the current session and feeds all results directly
// into the database. Auto-links persons to orgs.
// Uses crypto.decipher as primary cracker and MD5 dictionary
// as fallback (Viper's getpass approach).
// ============================================================

IntelManager = {}
IntelManager.mediator = null

// ---- Internal: crack a password hash ----

IntelManager.crackHash = function(username, hash)
    libManager = self.mediator.getExtension("libManager")
    crypto     = libManager.getCrypto()

    // Primary: crypto.decipher(username, hash)
    if not (crypto isa Error) then
        result = crypto.decipher(username, hash)
        if not (result isa Error) then
            return result
        end if
    end if

    // Fallback: MD5 dictionary comparison
    // md5(password) == hash
    dictionaryManager = self.mediator.getExtension("dictionaryManager")
    if dictionaryManager != null then
        password = dictionaryManager.get()
        while not (password isa Error)
            if md5(password) == hash then
                dictionaryManager.reset()
                return password
            end if
            password = dictionaryManager.get()
        end while
        dictionaryManager.reset()
    end if

    return Error.New("Could not crack hash for: " + username, "CRACK_FAILED")
end function

// ---- Internal: extract domain from email ----

IntelManager.domainFromEmail = function(email)
    if email.indexOf("@") == null then
        return ""
    end if
    return email[email.indexOf("@") + 1:]
end function

IntelManager.orgNameFromDomain = function(domain)
    parts = domain.split("\.")
    if parts.len == 0 then
        return domain
    end if
    return parts[0]
end function

// ---- Internal: ensure org exists, return org_id ----

IntelManager.ensureOrg = function(domain, publicIp)
    if domain.len == 0 then
        return ""
    end if
    dbManager = self.mediator.getExtension("databaseManager")
    orgName   = self.orgNameFromDomain(domain)
    orgData   = {}
    orgData.domain = domain
    orgData.name   = orgName
    orgData.ips    = [publicIp]
    return dbManager.upsertOrg(orgData)
end function

// ---- Harvest /etc/passwd ----

IntelManager.harvestPasswd = function(handler, publicIp, lanIp)
    dbManager = self.mediator.getExtension("databaseManager")
    results   = {"cracked": 0, "failed": 0, "skipped": 0}

    passwdFile = handler.getFile("/etc/passwd")
    if passwdFile isa Error then
        return Error.New("Could not read /etc/passwd", "NOT_FOUND")
    end if

    content = passwdFile.content()
    if content isa Error then
        return content
    end if

    lines = content.split(char(10))
    for line in lines
        if line.len == 0 then
            continue
        end if
        colonIdx = line.indexOf(":")
        if colonIdx == null then
            continue
        end if

        username = line[:colonIdx]
        hash     = line[colonIdx + 1:]
        if username.len == 0 or hash.len == 0 then
            continue
        end if

        password  = self.crackHash(username, hash)
        cracked   = not (password isa Error)

        credData = {}
        credData.type      = CRED_TYPES.SYSTEM
        credData.username  = username
        credData.hash      = hash
        credData.password  = ""
        credData.public_ip = publicIp
        credData.lan_ip    = lanIp
        credData.source    = "native"
        if cracked then
            credData.password = password
            results.cracked   = results.cracked + 1
        else
            results.failed = results.failed + 1
        end if

        dbManager.upsertCredential(credData)
    end for

    return results
end function

// ---- Harvest Mail.txt files ----

IntelManager.harvestMail = function(handler, publicIp, lanIp)
    dbManager = self.mediator.getExtension("databaseManager")
    results   = {"persons": 0, "passwords": 0, "failed": 0}

    homeDir = handler.getFile("/home")
    if homeDir isa Error then
        return homeDir
    end if

    kids = homeDir.children()
    if kids isa Error then
        return kids
    end if

    for userFolder in kids
        if userFolder.name() == "guest" then
            continue
        end if

        mailFile = handler.getFile("/home/" + userFolder.name() + "/Config/Mail.txt")
        if mailFile isa Error then
            continue
        end if
        content = mailFile.content()
        if content isa Error then
            continue
        end if

        lines     = content.split(char(10))
        name      = ""
        email     = ""
        domain    = ""
        netAddr   = ""

        for line in lines
            if line.len == 0 then
                continue
            end if

            // Email:hash credential line  -  format email@domain.com:hash
            colonIdx = line.indexOf(":")
            atIdx    = line.indexOf("@")
            if colonIdx != null and atIdx != null and atIdx < colonIdx then
                emailPart = line[:colonIdx]
                emailPart = emailPart.trim
                hashPart  = line[colonIdx + 1:]
                hashPart = hashPart.trim
                if emailPart.indexOf("@") != null and hashPart.len > 0 then
                    password = self.crackHash(emailPart, hashPart)
                    credData = {}
                    credData.type      = CRED_TYPES.MAIL
                    credData.username  = emailPart
                    credData.hash      = hashPart
                    credData.password  = ""
                    credData.public_ip = publicIp
                    credData.lan_ip    = lanIp
                    credData.source    = "native"
                    if not (password isa Error) then
                        credData.password     = password
                        results.passwords     = results.passwords + 1
                    else
                        results.failed = results.failed + 1
                    end if
                    dbManager.upsertCredential(credData)
                    // Use as primary email for this user if none found yet
                    if email.len == 0 then
                        email = emailPart
                    end if
                end if
                continue
            end if

            // From: Name <email> line
            if startsWith(line, "From:") then
                fromVal = line[5:]
                fromVal = fromVal.trim
                if fromVal.indexOf("<") != null and fromVal.indexOf(">") != null then
                    ltIdx = fromVal.indexOf("<")
                    gtIdx = fromVal.indexOf(">")
                    name  = fromVal[:ltIdx]
                    name = name.trim
                    email = fromVal[ltIdx + 1 : gtIdx]
                    email = email.trim
                else
                    email = fromVal.trim
                end if
                if email.indexOf("@") != null then
                    domain = self.domainFromEmail(email)
                end if
                continue
            end if

            // Name: line
            if startsWith(line, "Name:") then
                name = line[5:]
                name = name.trim
                continue
            end if

            // Network address
            trimmedLine = line.trim
            if is_valid_ip(trimmedLine) then
                netAddr = trimmedLine
                continue
            end if
        end for

        // Save person if we have an email
        if email.len > 0 then
            orgId = self.ensureOrg(domain, publicIp)

            personData = {}
            personData.name    = name
            personData.email   = email
            personData.org_id  = orgId
            personData.ips     = [publicIp]
            personData.lan_ips = [lanIp]
            personData.source  = "mail"
            if netAddr.len > 0 then
                personData.ips.push(netAddr)
            end if

            personId = dbManager.upsertPerson(personData)
            results.persons = results.persons + 1

            // Link credentials back to this person
            creds = dbManager.findCredentialsByIp(publicIp)
            for cred in creds
                if cred.username == email then
                    dbManager.linkCredentialToPerson(cred.id, personId)
                end if
            end for
        end if
    end for

    return results
end function

// ---- Harvest Bank.txt files ----

IntelManager.harvestBank = function(handler, publicIp, lanIp)
    dbManager = self.mediator.getExtension("databaseManager")
    results   = {"harvested": 0}

    homeDir = handler.getFile("/home")
    if homeDir isa Error then
        return homeDir
    end if

    kids = homeDir.children()
    if kids isa Error then
        return kids
    end if

    for userFolder in kids
        if userFolder.name() == "guest" then
            continue
        end if

        bankFile = handler.getFile("/home/" + userFolder.name() + "/Config/Bank.txt")
        if bankFile isa Error then
            continue
        end if
        content = bankFile.content()
        if content isa Error then
            continue
        end if

        bankStr = content.replace(char(10), " ").trim
        bankNum = bankStr
        bankPass = ""
        colonIdx = bankStr.indexOf(":")
        if colonIdx != null then
            bankNum = bankStr[:colonIdx]
            bankHash = bankStr[colonIdx + 1:]
            if bankHash.len > 0 then
                libManager = self.mediator.getExtension("libManager")
                cryptoLib  = libManager.getCrypto()
                if not (cryptoLib isa Error) then
                    deciphered = cryptoLib.decipher("", bankHash)
                    if not (deciphered isa Error) then
                        bankPass = deciphered
                    end if
                end if
            end if
        end if

        credData = {}
        credData.type        = CRED_TYPES.BANK
        credData.username    = userFolder.name()
        credData.password    = bankPass
        credData.hash        = ""
        credData.bank_number = bankNum
        credData.public_ip   = publicIp
        credData.lan_ip      = lanIp
        credData.source      = "native"

        dbManager.upsertCredential(credData)
        results.harvested = results.harvested + 1
    end for

    return results
end function

// ---- Harvest whois data ----

IntelManager.harvestWhois = function(ip)
    dbManager = self.mediator.getExtension("databaseManager")
    raw = whois(ip)
    if raw == null or raw == "No Info available" then
        return Error.New("No whois data for: " + ip, "NO_WHOIS")
    end if

    result = {}
    result.adminName  = ""
    result.adminEmail = ""
    result.domain     = ""
    result.phone      = ""

    lines = raw.split(char(10))
    for line in lines
        if line.indexOf("Administrative contact:") != null then
            adminSlice = line[line.indexOf(":") + 1:]
            result.adminName = adminSlice.trim
        end if
        if line.indexOf("Email address:") != null then
            adminSlice = line[line.indexOf(":") + 1:]
            result.adminEmail = adminSlice.trim
        end if
        if line.indexOf("Domain name:") != null then
            adminSlice = line[line.indexOf(":") + 1:]
            result.domain = adminSlice.trim
        end if
        if line.indexOf("Phone:") != null then
            adminSlice = line[line.indexOf(":") + 1:]
            result.phone = adminSlice.trim
        end if
    end for

    // Save org
    if result.domain.len > 0 then
        orgData = {}
        orgData.domain = result.domain
        orgData.name   = self.orgNameFromDomain(result.domain)
        orgData.ips    = [ip]
        orgId = dbManager.upsertOrg(orgData)

        // Save admin as a person
        if result.adminEmail.len > 0 then
            personData = {}
            personData.name    = result.adminName
            personData.email   = result.adminEmail
            personData.org_id  = orgId
            personData.ips     = [ip]
            personData.lan_ips = []
            personData.source  = "whois"
            dbManager.upsertPerson(personData)
        end if
    end if

    return result
end function

// ---- Full harvest on current session ----

IntelManager.harvestAll = function(publicIp, lanIp, forceSkipPasswd)
    sessionManager = self.mediator.getExtension("sessionManager")
    dbManager      = self.mediator.getExtension("databaseManager")
    handler        = sessionManager.currentSession.handler
    ui             = self.mediator.getExtension("corvidUi")
    priv           = handler.privilege()

    summary = {}
    summary.passwd   = {}
    summary.mail     = {}
    summary.bank     = {}
    summary.whois    = {}

    // ── Whois ─────────────────────────────────────────────────────────────
    // Skip if we already have org data for this IP
    existingOrgs = dbManager.findTarget(publicIp)
    hasWhois = false
    if existingOrgs.len > 0 then
        t = existingOrgs[0]
        if t.hasIndex("notes") and t.notes.indexOf("whois:done") != null then
            hasWhois = true
        end if
    end if
    if hasWhois then
        ui.dim("whois: already harvested - skipping")
    else
        ui.info("Running whois on " + publicIp + "...")
        whoisResult = self.harvestWhois(publicIp)
        if whoisResult isa Error then
            ui.warning("Whois: " + whoisResult.message)
        else
            summary.whois = whoisResult
            ui.success("Whois: " + whoisResult.domain + " | " + whoisResult.adminEmail)
            // Mark whois as done in target notes
            targets = dbManager.findTarget(publicIp)
            if targets.len > 0 then
                t2 = targets[0]
                existNotes = ""
                if t2.hasIndex("notes") then
                    existNotes = t2.notes
                end if
                if existNotes.indexOf("whois:done") == null then
                    if existNotes.len > 0 then
                        existNotes = existNotes + " "
                    end if
                    t2.notes = existNotes + "whois:done"
                    dbManager.upsertTarget(t2)
                end if
            end if
        end if
    end if

    // ── /etc/passwd ───────────────────────────────────────────────────────
    // Check what we already have vs what privilege can give us now
    // KEY: use LAN IP as primary key when available - different LAN machines
    // share the same public IP so checking by public IP causes false skips
    checkIp = publicIp
    if lanIp.len > 0 and lanIp != "unknown" and lanIp != publicIp then
        checkIp = lanIp
    end if
    existingCreds = dbManager.findCredentialsByIp(checkIp)
    hasRootHash   = false
    allUsersCracked = true
    crackedCount  = 0
    for ec in existingCreds
        if ec.type == "system" and ec.username == "root" and ec.password.len > 0 then
            hasRootHash = true
        end if
        if ec.type == "system" and ec.password.len == 0 then
            allUsersCracked = false
        end if
        if ec.type == "system" and ec.password.len > 0 then
            crackedCount = crackedCount + 1
        end if
    end for

    skipPasswd = false
    if forceSkipPasswd then
        skipPasswd = true
        ui.dim("passwd: skipping (handled by escalation strategy)")
    else if priv == "root" and hasRootHash and allUsersCracked then
        skipPasswd = true
        ui.dim("passwd: all hashes already cracked - skipping")
    else if priv != "root" and crackedCount > 0 then
        skipPasswd = true
        ui.dim("passwd: " + str(crackedCount) + " creds in DB, need root for complete harvest - skipping")
    end if

    if not skipPasswd then
        ui.info("Harvesting /etc/passwd...")
        passwdResult = self.harvestPasswd(handler, publicIp, lanIp)
        if passwdResult isa Error then
            ui.warning("passwd: " + passwdResult.message)
        else
            summary.passwd = passwdResult
            ui.success("passwd: " + passwdResult.cracked + " cracked, " + passwdResult.failed + " failed")
        end if
    end if

    // ── Mail files ────────────────────────────────────────────────────────
    // Always check by public IP - LAN IPs like 192.168.0.2 are shared across
    // many unrelated machines and cause false positive skips
    mailCreds = 0
    mailCredsForIp = dbManager.findCredentialsByIp(publicIp)
    for ec in mailCredsForIp
        if ec.type == "mail" then
            ecSource = ""
            if ec.hasIndex("source") then
                ecSource = ec.source
            end if
            if ecSource != "routerattack" and ecSource != "routerattack_file" then
                mailCreds = mailCreds + 1
            end if
        end if
    end for

    if mailCreds > 0 then
        ui.dim("mail: " + str(mailCreds) + " mail creds in DB - skipping")
    else
        ui.info("Harvesting mail files...")
        mailResult = self.harvestMail(handler, publicIp, lanIp)
        if mailResult isa Error then
            ui.warning("mail: " + mailResult.message)
        else
            summary.mail = mailResult
            ui.success("mail: " + mailResult.persons + " persons, " + mailResult.passwords + " passwords")
        end if
    end if

    // ── Bank files ────────────────────────────────────────────────────────
    // Always check by public IP for same reason as mail
    bankCreds = 0
    for ec in mailCredsForIp
        if ec.type == "bank" then
            ecSource = ""
            if ec.hasIndex("source") then
                ecSource = ec.source
            end if
            if ecSource != "routerattack" and ecSource != "routerattack_file" then
                bankCreds = bankCreds + 1
            end if
        end if
    end for

    if bankCreds > 0 then
        ui.dim("bank: " + str(bankCreds) + " records in DB - skipping")
    else
        ui.info("Harvesting bank files...")
        bankResult = self.harvestBank(handler, publicIp, lanIp)
        if bankResult isa Error then
            ui.warning("bank: " + bankResult.message)
        else
            summary.bank = bankResult
            ui.success("bank: " + bankResult.harvested + " records")
        end if
    end if

    // Save
    dbManager.save()
    ui.success("Database saved.")

    return summary
end function

// ---- Check bounce candidacy ----

IntelManager.checkBounceCandidate = function(publicIp, lanIp)
    sessionManager = self.mediator.getExtension("sessionManager")
    handler        = sessionManager.currentSession.handler

    // Check for SSH port
    ports = handler.getPorts()
    if ports isa Error then
        return false
    end if
    hasSsh = false
    for port in ports
        if port.port_number == 22 then
            hasSsh = true
        end if
    end for
    if not hasSsh then
        return false
    end if

    // Check for cracked root password in database
    dbManager = self.mediator.getExtension("databaseManager")
    creds = dbManager.findCredentialsByIp(publicIp)
    for cred in creds
        if cred.type == CRED_TYPES.SYSTEM and cred.username == "root" and cred.password.len > 0 then
            return cred.password
        end if
    end for

    return false
end function

IntelManager.setMediator = function(mediator)
    self.mediator = mediator
end function

IntelManager.New = function()
    result = new IntelManager
    return result
end function
