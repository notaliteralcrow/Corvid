// ============================================================
// CORVID 2.0 | managers/bounce/bounce-manager
// Manages bounce chain sessions.
// Bounce points are sessions tagged with bounce metadata.
// Handles chain building, verification, and cleanup.
// ============================================================

BOUNCE_TYPES = {}
BOUNCE_TYPES.OWNED       = "owned"
BOUNCE_TYPES.COMPROMISED = "compromised"

BounceManager = {}
BounceManager.mediator = null

BounceManager.getBounceTargets = function()
    dbManager = self.mediator.getExtension("databaseManager")
    return dbManager.findBounces()
end function

BounceManager.addBounce = function(publicIp, lanIp, port, username, password, bounceType, sourceIp)
    dbManager = self.mediator.getExtension("databaseManager")
    ui        = self.mediator.getExtension("corvidUi")

    if bounceType == null then
        bounceType = BOUNCE_TYPES.COMPROMISED
    end if

    // Avoid creating a duplicate target record. If a record for this public_ip
    // already exists, reuse its lan_ip so upsert's matchExisting succeeds and
    // we update the existing row instead of inserting a second one.
    existingForIp = dbManager.findTarget(publicIp)
    if existingForIp.len > 0 then
        rec0 = existingForIp[0]
        if rec0.hasIndex("lan_ip") and rec0.lan_ip.len > 0 then
            // Existing record has a lan_ip - reuse it
            lanIp = rec0.lan_ip
        end if
    end if

    portRecord = {}
    portRecord.port      = port
    portRecord.service   = "ssh"
    portRecord.version   = "unknown"
    portRecord.forwarded = true
    portRecord.exploited = true

    targetData = {}
    targetData.public_ip = publicIp
    targetData.lan_ip    = lanIp
    targetData.status    = TARGET_STATUS.BOUNCE
    targetData.ports     = [portRecord]
    // Don't append notes - existing record will keep its notes via upsert merge
    // Only set notes on new records (upsert handles merge)
    targetData.notes     = ""

    id = dbManager.upsertTarget(targetData)

    // Set the bounce marker only if not already present
    existing = dbManager.findTarget(publicIp)
    if existing.len > 0 then
        rec = existing[0]
        existingNotes = ""
        if rec.hasIndex("notes") then
            existingNotes = rec.notes
        end if
        if existingNotes.indexOf("bounce:" + bounceType) == null then
            dbManager.addNote("target", rec.id, "bounce:" + bounceType + " source:" + sourceIp)
        end if
    end if

    // Store SSH credentials. Use "owned" as source - "hardened" is reserved for
    // credentials that have actually been rotated by `bounces harden`.
    credData = {}
    credData.type      = "system"
    credData.username  = username
    credData.password  = password
    credData.public_ip = publicIp
    credData.lan_ip    = lanIp
    credData.source    = "owned"
    dbManager.upsertCredential(credData)
    dbManager.save()

    ui.success("Bounce point added: " + publicIp + " (" + bounceType + ")")
    return id
end function

BounceManager.listBounces = function()
    return self.getBounceTargets()
end function

BounceManager.verifyBounce = function(target)
    ui           = self.mediator.getExtension("corvidUi")
    dbManager    = self.mediator.getExtension("databaseManager")
    sessionManager = self.mediator.getExtension("sessionManager")

    // Find SSH credentials using helper
    sshCreds = dbManager.findRootSshCredentials(target.public_ip)
    sshPass = sshCreds.password
    sshNative = sshCreds.native_pass

    if sshPass.len == 0 and sshNative.len == 0 then
        ui.warning("No credentials found for: " + target.public_ip)
        return false
    end if

    sshPort = 22
    for port in target.ports
        if port.service == "ssh" then
            sshPort = port.port
        end if
    end for

    // Attempt connection - try current password then native_pass
    pivotShell = sessionManager.pivotSession.handler.shell
    result = null
    if sshPass.len > 0 then
        result = pivotShell.shell.connect_service(target.public_ip, sshPort, "root", sshPass, "ssh")
    end if
    if typeof(result) != "shell" and sshNative.len > 0 and sshNative != sshPass then
        result = pivotShell.shell.connect_service(target.public_ip, sshPort, "root", sshNative, "ssh")
    end if
    if typeof(result) == "shell" then
        ui.success("Bounce verified: " + target.public_ip)
        return true
    else
        ui.warning("Bounce unreachable: " + target.public_ip + "  -  " + str(result))
        return false
    end if
end function

BounceManager.verifyAll = function()
    bounces = self.getBounceTargets()
    if bounces.len == 0 then
        return Error.New("No bounce points configured", "EMPTY")
    end if
    results = {}
    for bTarget in bounces
        results[bTarget.public_ip] = self.verifyBounce(bTarget)
    end for
    return results
end function

BounceManager.connect = function(target)
    // Establish a session through a bounce point
    ui             = self.mediator.getExtension("corvidUi")
    dbManager      = self.mediator.getExtension("databaseManager")
    sessionManager = self.mediator.getExtension("sessionManager")

    sshCreds = dbManager.findRootSshCredentials(target.public_ip)
    sshPass = sshCreds.password
    sshNative = sshCreds.native_pass

    if sshPass.len == 0 and sshNative.len == 0 then
        return Error.New("No credentials for bounce point: " + target.public_ip, "NO_CREDS")
    end if

    sshPort = 22
    for port in target.ports
        if port.service == "ssh" then
            sshPort = port.port
        end if
    end for

    currentShell = sessionManager.currentSession.handler.shell
    newShell = null
    if sshPass.len > 0 then
        newShell = currentShell.shell.connect_service(target.public_ip, sshPort, "root", sshPass, "ssh")
    end if
    if typeof(newShell) != "shell" and sshNative.len > 0 and sshNative != sshPass then
        newShell = currentShell.shell.connect_service(target.public_ip, sshPort, "root", sshNative, "ssh")
    end if

    if typeof(newShell) != "shell" then
        return Error.New("Could not connect to bounce: " + target.public_ip + "  -  " + str(newShell), "CONNECT_ERROR")
    end if

    newSession = sessionManager.addSession(newShell, "bounce@" + target.public_ip)
    if newSession isa Error then
        return newSession
    end if
    sessionManager.setCurrentSession(newSession)
    ui.success("Connected through bounce: " + target.public_ip)
    return newSession
end function

BounceManager.cleanBounce = function(target)
    // Clean logs on a bounce machine and disconnect
    ui             = self.mediator.getExtension("corvidUi")
    sessionManager = self.mediator.getExtension("sessionManager")

    notes = target.notes
    bounceType = BOUNCE_TYPES.COMPROMISED
    if notes.indexOf("bounce:owned") != null then
        bounceType = BOUNCE_TYPES.OWNED
    end if

    if bounceType == BOUNCE_TYPES.OWNED then
        // Owned  -  wipe all logs freely
        handler = sessionManager.currentSession.handler
        result  = handler.corruptLogs()
        if result isa Error then
            ui.warning("Log wipe failed: " + result.message)
        else
            ui.success("Logs wiped on owned bounce: " + target.public_ip)
        end if
    else
        // Compromised  -  manual checklist only
        ui.print("")
        ui.info("Bounce cleanup checklist for COMPROMISED machine " + target.public_ip + ":")
        ui.print("  1. cd /var")
        ui.print("  2. LogViewer.exe system.log")
        ui.print("  3. Delete: shell obtained, connection routed entries")
        ui.print("  4. Keep: connect and disconnect entries")
        ui.print("  5. Verify nothing else is suspicious")
        ui.print("")
    end if
end function

BounceManager.removeBounce = function(publicIp)
    dbManager = self.mediator.getExtension("databaseManager")
    ui        = self.mediator.getExtension("corvidUi")

    // CRITICAL: use findBounces + filter by IP, NOT findTarget()
    // findTarget uses ipIndex which only holds ONE record per public IP -
    // it would return the router record, not the bounce record
    allBounces = dbManager.findBounces()
    targets = []
    for b in allBounces
        if b.hasIndex("public_ip") and b.public_ip == publicIp then
            targets.push(b)
        end if
    end for

    if targets.len == 0 then
        return Error.New("Bounce not found: " + publicIp, "NOT_FOUND")
    end if

    // Update each matching bounce record directly via targetDb
    for target in targets
        target.status = TARGET_STATUS.OWNED
        existingNotes = ""
        if target.hasIndex("notes") then
            existingNotes = target.notes
        end if
        if existingNotes.indexOf("removed_bounce") == null then
            if existingNotes.len > 0 then
                existingNotes = existingNotes + char(10)
            end if
            target.notes = existingNotes + "removed_bounce"
        end if
        dbManager.targetDb.db.set(target.id, target)
        dbManager.targetDb.db.dirty = true
    end for

    dbManager.save()
    ui.success("Bounce removed: " + publicIp)
end function

BounceManager.findBounce = function(publicIp)
    bounces = self.getBounceTargets()
    for b in bounces
        if b.hasIndex("public_ip") and b.public_ip == publicIp then
            return b
        end if
    end for
    return null
end function

BounceManager.setStatus = function(publicIp, status)
    dbManager = self.mediator.getExtension("databaseManager")
    bounces = self.getBounceTargets()
    for b in bounces
        if b.hasIndex("public_ip") and b.public_ip == publicIp then
            // Store health in notes, NOT in status
            // status must stay "bounce" so findBounces() keeps finding this record
            existingNotes = ""
            if b.hasIndex("notes") then
                existingNotes = b.notes
            end if
            // Replace health marker - use direct DB set to avoid upsert status protection
            newNotes = ""
            for part in existingNotes.split(" ")
                trimPart = part.trim
                if trimPart.len > 0 and trimPart.indexOf("health:") != 0 then
                    if newNotes.len > 0 then
                        newNotes = newNotes + " "
                    end if
                    newNotes = newNotes + trimPart
                end if
            end for
            if newNotes.len > 0 then
                newNotes = newNotes + " "
            end if
            newNotes = newNotes + "health:" + status
            b.notes = newNotes
            dbManager.targetDb.db.set(b.id, b)
            dbManager.targetDb.db.dirty = true
            dbManager.save()
            return true
        end if
    end for
    return false
end function

BounceManager.getHealth = function(b)
    if not b.hasIndex("notes") then
        return "unknown"
    end if
    for part in b.notes.split(" ")
        if part.indexOf("health:") == 0 then
            return part[7:]
        end if
    end for
    return "unknown"
end function

BounceManager.setMediator = function(mediator)
    self.mediator = mediator
end function

BounceManager.New = function()
    result = new BounceManager
    return result
end function
