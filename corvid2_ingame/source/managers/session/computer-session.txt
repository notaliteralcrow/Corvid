// ============================================================
// CORVID 2.0 | managers/session/computer-session
// Extends FileSession with computer-specific operations:
// user management, process control, file creation.
// ============================================================

ComputerSession = new FileSession

ComputerSession.computer = null

ComputerSession.ip = function()
    result = {}
    result.public = self.computer.computer.public_ip
    result.local  = self.computer.lanIp()
    return result
end function

ComputerSession.name = function()
    return self.computer.name()
end function

ComputerSession.getFile = function(path)
    if path == null or path.len == 0 then
        return self.current
    end if
    // Absolute path - try computer.File() first
    if startsWith(path, "/") then
        raw = self.computer.getFile(path)
        if not (raw isa Error) then
            return raw
        end if
    end if
    // Relative path - traverse manually
    startingPoint = self.current
    parts = path.split("/")
    if startsWith(path, "/") then
        startingPoint = self.root
        parts = parts[1:]
    end if
    for part in parts
        if part == ".." then
            parent = startingPoint.parent()
            if parent isa Error then
                return Error.New("Cannot go above root", "AT_ROOT")
            end if
            startingPoint = parent
        else if part == "." or part.len == 0 then
            continue
        else
            child = startingPoint.child(part)
            if child isa Error then
                return Error.New("Not found: " + path, "NOT_FOUND")
            end if
            startingPoint = child
        end if
    end for
    return startingPoint
end function

ComputerSession.createFolder = function(path)
    parts  = path.split("/")
    name   = parts.pop()
    parent = self.getFile(parts.join("/"))
    if parent isa Error then
        return parent
    end if
    return self.computer.createFolder(parent.path(), name)
end function

ComputerSession.createFile = function(path)
    parts  = path.split("/")
    name   = parts.pop()
    parent = self.getFile(parts.join("/"))
    if parent isa Error then
        return parent
    end if
    return self.computer.touch(parent.path(), name)
end function

ComputerSession.addUser = function(user, password)
    return self.computer.addUser(user, password)
end function

ComputerSession.deleteUser = function(user)
    return self.computer.deleteUser(user)
end function

ComputerSession.changePassword = function(user, password)
    return self.computer.changePassword(user, password)
end function

ComputerSession.showProcs = function()
    return self.computer.showProcs()
end function

ComputerSession.closeProcess = function(pid)
    return self.computer.closeProcess(pid)
end function

ComputerSession.getPorts = function()
    return self.computer.getPorts()
end function

ComputerSession.hasActiveUsers = function()
    procs = self.computer.showProcs()
    if procs isa Error then
        return false
    end if
    for proc in procs
        if proc.hasIndex("COMMAND") then
            if proc.COMMAND == "dsession" then
                return true
            end if
        end if
    end for
    return false
end function

ComputerSession.type = function()
    return "computer"
end function

ComputerSession.New = function(computer)
    result        = new ComputerSession
    result.computer = computer
    rootFile      = computer.getFile("/")
    if rootFile isa Error then
        result.root    = null
        result.current = null
    else
        result.root    = climbFileToRoot(rootFile)
        result.current = rootFile
    end if
    result.user = getUserFromHandler(result)
    return result
end function
