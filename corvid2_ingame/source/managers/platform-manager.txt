// ============================================================
// CORVID 2.0 | managers/platform-manager
// Manages attack platform mode. When active, exploit traffic
// originates from a roost node rather than the Nest.
// Corvid continues running on Nest - DB, sessions, and UI
// are all unaffected. Only net_use calls are rerouted.
//
// Requires: metaxploit.so and crypto.so on the roost node.
// If not present, they are uploaded from Nest automatically.
// Tracks what was uploaded so wipe() removes only those files.
// ============================================================

PlatformManager = {}
PlatformManager.mediator        = null
PlatformManager.active          = false
PlatformManager.nodeIp          = ""    // IP of the active platform node
PlatformManager.roostShell      = null  // raw shell on node
PlatformManager.roostMeta       = null  // Metaxploit facade loaded from node
PlatformManager.roostCrypto     = null  // Crypto facade loaded from node
PlatformManager.uploadedMeta    = false // true if we had to upload metaxploit.so
PlatformManager.uploadedCrypto  = false // true if we had to upload crypto.so
PlatformManager.metaPath        = ""    // path to metaxploit.so on node
PlatformManager.cryptoPath      = ""    // path to crypto.so on node

// ── activate ────────────────────────────────────────────────────────────────
// Activates platform mode on the current session. The current session's
// machine becomes the attack origin for all net_use calls.
// Use roost pick, bounces chain, or flock adopt first to position yourself
// on the desired node, then run platform deploy.

PlatformManager.activate = function()
    ui             = self.mediator.getExtension("corvidUi")
    sessionManager = self.mediator.getExtension("sessionManager")
    pivotSession   = sessionManager.pivotSession
    pivotHandler   = pivotSession.handler
    pivotShell     = pivotHandler.shell.shell
    currentSession = sessionManager.currentSession

    if self.active then
        ui.warning("Platform already deployed on " + self.nodeIp + " - run: platform wipe  or  platform sleep  first")
        return Error.New("Already active", "ALREADY_ACTIVE")
    end if

    if currentSession == pivotSession then
        return Error.New("Cannot deploy platform on Nest - switch to a remote session first", "IS_NEST")
    end if

    handler    = currentSession.handler
    nodeShell  = handler.shell.shell
    nodeComp   = nodeShell.host_computer
    nodeIp     = currentSession.data.ip.public

    ui.info("Activating platform on current session: " + nodeIp + "...")

    // ── Ensure /shiny exists on node ─────────────────────────────────────
    shinyDir = nodeComp.File("/shiny")
    if shinyDir == null then
        nodeComp.create_folder("/", "shiny")
    end if

    // ── Ensure metaxploit.so is available ────────────────────────────────
    metaFile = nodeComp.File("/lib/metaxploit.so")
    if metaFile == null then
        metaFile = nodeComp.File("/usr/lib/metaxploit.so")
    end if

    if metaFile == null then
        ui.info("  metaxploit.so not found on node - uploading from Nest...")
        nestMeta = pivotHandler.getFile("/lib/metaxploit.so")
        if nestMeta isa Error then
            return Error.New("metaxploit.so not found on Nest either", "NO_META")
        end if
        scpMeta = pivotShell.scp("/lib/metaxploit.so", "/lib", nodeShell)
        if typeof(scpMeta) == "string" and scpMeta.len > 0 then
            scpMeta = pivotShell.scp("/lib/metaxploit.so", "/shiny", nodeShell)
            if typeof(scpMeta) == "string" and scpMeta.len > 0 then
                return Error.New("Could not upload metaxploit.so: " + scpMeta, "UPLOAD_FAILED")
            end if
            self.metaPath = "/shiny/metaxploit.so"
        else
            self.metaPath = "/lib/metaxploit.so"
        end if
        self.uploadedMeta = true
        ui.success("  Uploaded metaxploit.so -> " + self.metaPath)
    else
        self.metaPath = metaFile.path
        self.uploadedMeta = false
        ui.dim("  metaxploit.so found at " + self.metaPath)
    end if

    // ── Ensure crypto.so is available ────────────────────────────────────
    cryptoFile = nodeComp.File("/lib/crypto.so")
    if cryptoFile == null then
        cryptoFile = nodeComp.File("/usr/lib/crypto.so")
    end if

    if cryptoFile == null then
        ui.info("  crypto.so not found on node - uploading from Nest...")
        nestCrypto = pivotHandler.getFile("/lib/crypto.so")
        if nestCrypto isa Error then
            ui.warning("  crypto.so not found on Nest - some operations may be limited")
            self.cryptoPath = ""
            self.uploadedCrypto = false
        else
            scpCrypto = pivotShell.scp("/lib/crypto.so", "/lib", nodeShell)
            if typeof(scpCrypto) == "string" and scpCrypto.len > 0 then
                scpCrypto = pivotShell.scp("/lib/crypto.so", "/shiny", nodeShell)
                if typeof(scpCrypto) == "string" and scpCrypto.len > 0 then
                    ui.warning("  Could not upload crypto.so - some operations may be limited")
                    self.cryptoPath = ""
                    self.uploadedCrypto = false
                else
                    self.cryptoPath = "/shiny/crypto.so"
                    self.uploadedCrypto = true
                    ui.success("  Uploaded crypto.so -> " + self.cryptoPath)
                end if
            else
                self.cryptoPath = "/lib/crypto.so"
                self.uploadedCrypto = true
                ui.success("  Uploaded crypto.so -> " + self.cryptoPath)
            end if
        end if
    else
        self.cryptoPath = cryptoFile.path
        self.uploadedCrypto = false
        ui.dim("  crypto.so found at " + self.cryptoPath)
    end if

    // ── Load metaxploit from node ─────────────────────────────────────────
    rawMeta = include_lib(self.metaPath)
    if rawMeta == null then
        return Error.New("include_lib failed for " + self.metaPath, "LIB_LOAD_FAILED")
    end if
    self.roostMeta = Metaxploit.New(rawMeta)
    if self.roostMeta isa Error then
        return Error.New("Could not wrap node metaxploit: " + self.roostMeta.message, "LIB_WRAP_FAILED")
    end if

    // ── Load crypto from node if available ───────────────────────────────
    if self.cryptoPath.len > 0 then
        rawCrypto = include_lib(self.cryptoPath)
        if rawCrypto != null then
            self.roostCrypto = Crypto.New(rawCrypto)
        end if
    end if

    self.active    = true
    self.nodeIp    = nodeIp
    self.roostShell = nodeShell

    ui.success("Platform deployed - attacks will originate from " + nodeIp)
    return null
end function

// ── deactivate ───────────────────────────────────────────────────────────────

PlatformManager.deactivate = function()
    ui = self.mediator.getExtension("corvidUi")
    if not self.active then
        ui.warning("Platform not active")
        return
    end if
    nodeIp = self.nodeIp
    self.active         = false
    self.nodeIp         = ""
    self.roostShell     = null
    self.roostMeta      = null
    self.roostCrypto    = null
    self.metaPath       = ""
    self.cryptoPath     = ""
    self.uploadedMeta   = false
    self.uploadedCrypto = false
    ui.success("Platform sleeping - attacks returning to Nest context")
end function

// ── wipe ─────────────────────────────────────────────────────────────────────
// Remove only what platform uploaded. Never deletes native libs or touches
// anything else on the node. Node stays healthy for future use.

PlatformManager.wipe = function()
    ui = self.mediator.getExtension("corvidUi")
    if not self.active then
        ui.warning("Platform not active - nothing to wipe")
        return
    end if

    roostComp = self.roostShell.host_computer
    wiped = []

    if self.uploadedMeta and self.metaPath.len > 0 then
        f = roostComp.File(self.metaPath)
        if f != null then
            f.delete()
            wiped.push(self.metaPath)
        end if
    end if

    if self.uploadedCrypto and self.cryptoPath.len > 0 then
        f = roostComp.File(self.cryptoPath)
        if f != null then
            f.delete()
            wiped.push(self.cryptoPath)
        end if
    end if

    if wiped.len > 0 then
        for w in wiped
            ui.success("  Removed: " + w)
        end for
    else
        ui.dim("  Nothing to remove (no files were uploaded)")
    end if

    // Corrupt logs on the platform node before deactivating
    // Uses the move trick: create empty file, move onto system.log
    ui.info("  Wiping logs on platform node...")
    logFile = roostComp.File("/var/system.log")
    if logFile == null then
        ui.dim("  Log file not found - skipping")
    else
        roostComp.touch("/var", "t")
        emptyFile = roostComp.File("/var/t")
        if emptyFile != null then
            moveResult = emptyFile.move("/var", "system.log")
            if moveResult == 1 then
                ui.success("  Logs wiped")
            else
                emptyFile.delete()
                ui.dim("  Log wipe failed - may need root access")
            end if
        end if
    end if

    // Clean up node-attacker, node-router, and job/results files
    cleanNames = ["node-attacker", "node-router", "corvid-job.json", "corvid-results.json", "corvid-router-job.json", "corvid-router-results.json"]
    for cleanName in cleanNames
        cf = roostComp.File("/shiny/" + cleanName)
        if cf != null then
            cf.delete()
            ui.success("  Removed: /shiny/" + cleanName)
        end if
    end for

    self.deactivate()
end function

// ── getters ──────────────────────────────────────────────────────────────────

PlatformManager.getMeta = function()
    if self.active and self.roostMeta != null then
        return self.roostMeta
    end if
    return null
end function

PlatformManager.getCrypto = function()
    if self.active and self.roostCrypto != null then
        return self.roostCrypto
    end if
    return null
end function

PlatformManager.status = function()
    if not self.active then
        return "inactive"
    end if
    return self.nodeIp
end function

PlatformManager.setMediator = function(mediator)
    self.mediator = mediator
end function

PlatformManager.New = function()
    result = new PlatformManager
    result.active         = false
    result.nodeIp         = ""
    result.roostShell     = null
    result.roostMeta      = null
    result.roostCrypto    = null
    result.uploadedMeta   = false
    result.uploadedCrypto = false
    result.metaPath       = ""
    result.cryptoPath     = ""
    return result
end function
