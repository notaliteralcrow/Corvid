RoostManager = {}
RoostManager.mediator = null
RoostManager.db = null

ROOST_PROCESS_NAMES = ["stick", "shiny", "grub", "wing"]

RoostManager.init = function()
    self.db = RoostDb
    self.db.setMediator(self.mediator)
end function

RoostManager.randomProcessName = function()
    idx = floor(time * 13) % ROOST_PROCESS_NAMES.len
    return ROOST_PROCESS_NAMES[idx]
end function

RoostManager.list = function()
    return self.db.records
end function

RoostManager.listActive = function()
    return self.db.listActive()
end function

RoostManager.add = function(publicIp, lanIp, processName)
    return self.db.add(publicIp, lanIp, processName)
end function

RoostManager.remove = function(publicIp)
    return self.db.remove(publicIp)
end function

RoostManager.find = function(publicIp)
    return self.db.find(publicIp)
end function

RoostManager.setStatus = function(publicIp, status)
    self.db.setStatus(publicIp, status)
end function

RoostManager.markUsed = function(publicIp)
    self.db.markUsed(publicIp)
end function

RoostManager.pickLeastUsed = function()
    return self.db.pickLeastUsed()
end function

RoostManager.load = function(data)
    self.db.load(data)
end function

RoostManager.save = function(data)
    self.db.save(data)
end function

RoostManager.setMediator = function(mediator)
    self.mediator = mediator
end function

RoostManager.New = function()
    result = new RoostManager
    return result
end function

// ── adoptNode ────────────────────────────────────────────────
// Shared adoption logic used by both roost pick and roost adopt.
// Finds the flock connection, creates a session, loads node metaxploit.
RoostManager.adoptNode = function(node, mediator)
    ui             = mediator.getExtension("corvidUi")
    sessionManager = mediator.getExtension("sessionManager")
    dbManager      = mediator.getExtension("databaseManager")
    libManager     = mediator.getExtension("libManager")

    ui.info("Connecting to: " + node.public_ip + " (process: " + node.process_name + ")")

    // Find this node in active flock connections
    meta = include_lib("/lib/metaxploit.so")
    if meta == null then
        return Error.New("metaxploit.so not available", "NO_META")
    end if

    shells = meta.rshell_server()
    if shells == null or shells.len == 0 then
        return Error.New("No active flock connections - node may be dead", "NO_RSHELL")
    end if

    targetShell = null
    for sh in shells
        if typeof(sh) != "shell" then
            continue
        end if
        comp = sh.host_computer
        if comp != null then
            if comp.public_ip == node.public_ip and comp.local_ip == node.lan_ip then
                targetShell = sh
                break
            end if
        end if
    end for

    if targetShell == null then
        // Fallback: match on public IP only, prefer root
        for sh in shells
            if typeof(sh) != "shell" then
                continue
            end if
            comp = sh.host_computer
            if comp != null then
                if comp.public_ip == node.public_ip then
                    rootDir = sh.host_computer.File("/root")
                    if rootDir != null and rootDir.has_permission("w") then
                        targetShell = sh
                        break
                    end if
                end if
            end if
        end for
    end if

    if targetShell == null then
        dbManager.roostDb.setStatus(node.public_ip, ROOST_STATUS.DEAD)
        dbManager.save()
        return Error.New("Could not find flock connection for " + node.public_ip + " - marked dead", "NO_SHELL")
    end if

    // Adopt as session
    newSession = sessionManager.addSession(targetShell, "roost@" + node.public_ip)
    if newSession isa Error then
        return newSession
    end if
    sessionManager.setCurrentSession(newSession)
    dbManager.roostDb.markUsed(node.public_ip)
    dbManager.save()

    // Load roost node's local metaxploit as current lib
    ui.info("Loading roost node metaxploit library...")
    pivotHandler = sessionManager.pivotSession.handler
    pivotShell   = pivotHandler.shell.shell

    nodeMetaFile = targetShell.host_computer.File("/lib/metaxploit.so")
    if nodeMetaFile != null then
        scpDest = "/shiny/metaxploit.so"
        scpErr  = pivotShell.scp("/lib/metaxploit.so", "/shiny", targetShell)
        if typeof(scpErr) != "string" or scpErr.len == 0 then
            nestComp    = pivotShell.host_computer
            rawNodeMeta = include_lib(scpDest)
            if rawNodeMeta != null then
                nodeIpData       = {}
                nodeIpData.ip    = {"public": node.public_ip, "local": node.lan_ip}
                nodeIpData.path  = scpDest
                nodeLib = libManager.addLib(rawNodeMeta, nodeIpData)
                if not (nodeLib isa Error) then
                    libManager.setCurrentLib(nodeLib)
                    ui.success("Roost node metaxploit loaded (v" + rawNodeMeta.version + ")")
                else
                    ui.dim("  Could not register node lib: " + nodeLib.message)
                end if
            else
                ui.dim("  include_lib failed for node metaxploit - using Nest lib")
            end if
            cleanFile = nestComp.File(scpDest)
            if cleanFile != null then
                cleanFile.delete()
            end if
        else
            ui.dim("  Could not SCP node metaxploit - using Nest lib")
        end if
    else
        ui.dim("  Node has no /lib/metaxploit.so - using Nest lib")
    end if

    ui.success("Now on roost node: " + node.public_ip)
    ui.info("Run attack commands from here. All connections originate from this node.")
    ui.separator()
end function
