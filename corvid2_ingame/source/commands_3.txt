// --- attack ---
_cmd = new Command
_cmd.name = "attack"
_cmd.description = "Guided attack workflow. Recon, exploit, pivot to LAN target, harvest, report. Usage: attack [public_ip] (lan_ip)"
_cmd.parameters = [StringParameter.New({"name": "publicip", "required": true}), StringParameter.New({"name": "lanip", "required": false, "default": ""})]
_cmd.permissions = {}
_cmd.permissions.object = "shell"
_cmd.permissions.privilege = "guest"
_cmd.callback = function(params, mediator)
    ui             = mediator.getExtension("corvidUi")
    sessionManager = mediator.getExtension("sessionManager")
    dbManager      = mediator.getExtension("databaseManager")
    attackManager  = mediator.getExtension("attackManager")
    intelManager   = mediator.getExtension("intelManager")
    metaxManager   = mediator.getExtension("metaxploitManager")
    libManager     = mediator.getExtension("libManager")

    publicIp  = params.publicip
    lanTarget = params.lanip
    hasLan = lanTarget.len > 0

    // ── helper: privilege rank for comparison ────────────────────────────
    privRank = function(priv)
        if priv == "root" then
            return 3
        end if
        if priv == "user" then
            return 2
        end if
        if priv == "guest" then
            return 1
        end if
        return 0
    end function

    // ── helper: pick best exploit result from a list ─────────────────────
    bestResult = function(results)
        best = null
        for r in results
            if best == null then
                best = r
            else if r.rtype == "shell" and best.rtype != "shell" then
                best = r
            else if r.rtype == "shell" and best.rtype == "shell" then
                if privRank(r.privilege) > privRank(best.privilege) then
                    best = r
                end if
            else if r.rtype == "computer" and best.rtype == "file" then
                best = r
            end if
        end for
        return best
    end function

    // ── helper: ask user what to do when stuck ───────────────────────────
    askStuck = function(context, priv, ip)
        ui.print("")
        ui.warning("Got " + priv + " access on " + ip + " (" + context + ")")
        ui.print("  1. Harvest credentials and attempt escalate")
        ui.print("  2. Continue to next phase with current access")
        ui.print("  3. Try exploits manually (msfconsole)")
        ui.print("  4. Stop and report what we have")
        ui.print("")
        answer = user_input("Choose [1-4]: ")
        return answer.trim
    end function


    // ── helper: clean a single session then step back ───────────────────
    cleanOneSession = function(leaveMode)
        sm      = mediator.getExtension("sessionManager")
        cfg     = mediator.getExtension("configManager")
        h       = sm.currentSession.handler
        ip      = sm.currentSession.data.ip
        label   = sm.currentSession.label

        ui.info("Cleaning: " + label + " (" + ip.public + ")")

        // Dead-session check - rebooted or interrupted connections will fail here
        priv = "unknown"
        privResult = h.privilege()
        if typeof(privResult) == "string" then
            priv = privResult
        else
            // Session is dead - remove it directly without cleanup
            ui.warning("  Session unreachable (rebooted or disconnected) - skipping cleanup")
            deadSession = sm.currentSession
            deadIdx = sm.getSessionIndex(deadSession)
            if deadIdx != null and deadIdx > 0 then
                sm.sessions.remove(deadIdx)
                newChainD = []
                for scd in sm.sessionChain
                    if scd != deadSession then
                        newChainD.push(scd)
                    end if
                end for
                sm.sessionChain = newChainD
            end if
            if sm.sessions.len > 0 then
                sm.currentSession = sm.sessions[sm.sessions.len - 1]
            end if
            return
        end if

        // Kill and remove AdminMonitor if active on this IP
        activeMonIp = cfg.get("monitorActive")
        if not (activeMonIp isa Error) and activeMonIp == ip.public then
            procs = h.showProcs()
            if not (procs isa Error) then
                for proc in procs
                    if proc.hasIndex("COMMAND") then
                        if proc.COMMAND == "AdminMonitor.exe" or proc.COMMAND == "AdminMonitor.exe" then
                            pidV = ""
                            if proc.hasIndex("PID") then
                                pidV = proc.PID
                            end if
                            if pidV.len > 0 then
                                h.closeProcess(pidV.val)
                                ui.success("  AdminMonitor process killed.")
                            end if
                        end if
                    end if
                end for
            end if
            tmpF = h.getFile("/root/AdminMonitor.exe")
            if not (tmpF isa Error) then
                tmpF.delete()
                ui.success("  /root/AdminMonitor.exe deleted.")
            end if
            cfg.set("monitorActive", "")
        end if

        // Delete cnmap binary and output if we ran lannmap on this session
        activeCnmapIp = cfg.get("cnmapActive")
        if not (activeCnmapIp isa Error) and activeCnmapIp == ip.public then
            cnmapF = h.getFile("/root/cnmap")
            if not (cnmapF isa Error) then
                cnmapF.delete()
                ui.success("  /root/cnmap deleted.")
            end if
            cnmapOutF = h.getFile("/root/cnmap_out.txt")
            if not (cnmapOutF isa Error) then
                cnmapOutF.delete()
                ui.success("  /root/cnmap_out.txt deleted.")
            end if
            cfg.set("cnmapActive", "")
        end if

        // Delete any other files we uploaded to /shiny/ including scp'd .so files from local lib scan
        tmpDir = h.getFile("/shiny")
        if not (tmpDir isa Error) then
            tmpKids = tmpDir.children()
            if not (tmpKids isa Error) then
                for tmpFile in tmpKids
                    tName = tmpFile.name()
                    if tName != "." and tName != ".." then
                        if tmpFile.owner() == h.user then
                            tmpFile.delete()
                            ui.success("  Deleted /shiny/" + tName)
                        end if
                    end if
                end for
            end if
        end if
        // Also clean Nest /tmp of any .so files left by local lib scan
        // (these are on session 0, not the target - clean regardless)
        nestSession = sm.sessions[0]
        if nestSession != null then
            nestComp = nestSession.handler.shell.shell.host_computer
            nestTmp = nestComp.File("/shiny")
            if nestTmp != null then
                nestTmpFiles = nestTmp.get_files
                if nestTmpFiles != null then
                    for ntf in nestTmpFiles
                        if ntf.name.indexOf(".so") != null then
                            ntf.delete()
                            ui.success("  Deleted Nest /shiny/" + ntf.name)
                        end if
                    end for
                end if
            end if
        end if

        // If not root, auto-attempt escalation before log wipe
        escalated2 = false
        if priv != "root" then
            ui.info("  Not root - attempting auto-escalate before log wipe...")
            atkMgr = mediator.getExtension("attackManager")
            esResult = atkMgr.escalate()
            if not (esResult isa Error) then
                priv  = sm.currentSession.handler.privilege()
                h     = sm.currentSession.handler
                escalated2 = true
                ui.success("  Escalated to root for cleanup.")
            else
                ui.warning("  Could not escalate. Log cannot be wiped on this session.")
            end if
        end if

        // Handle log corruption
        if priv == "root" then
            wipeResult = h.corruptLogs()
            if not (wipeResult isa Error) then
                ui.success("  Log wiped.")
            else
                ui.warning("  Log wipe failed: " + wipeResult.message)
            end if
        else
            ui.warning("  Continuing without log wipe.")
        end if

        // Remove current session from list and switch to last remaining non-current session
        // Don't use back() since sessionChain gets corrupted by 'use' calls
        leavingSession = sm.currentSession
        leavingIdx = sm.getSessionIndex(leavingSession)
        if leavingIdx != null and leavingIdx > 0 then
            sm.sessions.remove(leavingIdx)
            // Also remove from sessionChain to keep it clean
            newChain = []
            for sc in sm.sessionChain
                if sc != leavingSession then
                    newChain.push(sc)
                end if
            end for
            sm.sessionChain = newChain
        end if
        // If escalated, also remove the original guest session
        if escalated2 then
            ui.info("  Closing original guest session...")
            // The escalated session was added on top - remove it too
            if sm.sessions.len > 1 then
                escalatedIdx = sm.sessions.len - 1
                escalatedSess = sm.sessions[escalatedIdx]
                sm.sessions.remove(escalatedIdx)
                newChain2 = []
                for sc2 in sm.sessionChain
                    if sc2 != escalatedSess then
                        newChain2.push(sc2)
                    end if
                end for
                sm.sessionChain = newChain2
            end if
        end if
        // Set current to last remaining session
        if sm.sessions.len > 0 then
            sm.currentSession = sm.sessions[sm.sessions.len - 1]
            sm.sessionChain = [sm.sessions[0]]
        end if
        ui.success("  Backed out to: " + sm.currentSession.label)
    end function

    // ── helper: full leave - clean every session back to Nest ─────────────
    doLeave = function(leaveMode)
        sm = mediator.getExtension("sessionManager")

        // Count how many sessions deep we are (session 0 is Nest - never clean it)
        // Walk back through every session until we reach the Nest
        ui.separator()
        ui.info("Starting full cleanup sequence (" + leaveMode + " mode)...")
        ui.print("")

        maxSteps = 20 // safety limit
        step = 0
        while step < maxSteps
            sessions = sm.listSessions()
            if sessions.len <= 1 then
                break // only Nest remains
            end if
            // Always switch to the LAST non-Nest session in the list
            // This ignores sessionChain order entirely so 'use' mid-chain doesn't break leave
            lastIdx = sessions.len - 1
            lastSession = sessions[lastIdx]
            if lastSession.label == "Nest" then
                break
            end if
            sm.setCurrentSession(lastSession)
            currentLabel = sm.currentSession.label
            cleanOneSession(leaveMode)
            newLabel = sm.currentSession.label
            if newLabel == currentLabel then
                ui.warning("  Session did not change after cleanup - breaking to avoid infinite loop.")
                break
            end if
            step = step + 1
        end while

        ui.print("")
        ui.success("Full cleanup complete. Back on Nest.")
        ui.separator()
    end function

        result = intelManager.harvestAll(pubIp, lanIp)

        end if
    end function

    // ── helper: try escalate ─────────────────────────────────────────────
    tryEscalate = function()
        ui.info("Attempting escalate...")
        result = attackManager.escalate()
        if result isa Error then
            ui.warning(result.message)
            return false
        end if
        return true
    end function

    // ── helper: scan all ports, return best overall shell ────────────────
    scanAllPorts = function(pubIp, lanIp, routerIp, portList)
        allResults  = {}
        bestShell   = null
        bestPriv    = 0

        for portRecord in portList
            if not portRecord.forwarded then
                continue
            end if
            port = portRecord.port
            if port == 0 then
                continue
            end if

            ui.info("Scanning port " + str(port) + " (" + portRecord.service + " " + portRecord.version + ")...")
            results = attackManager.runScan(pubIp, port, lanIp, routerIp)
            if results isa Error then
                ui.warning("Port " + str(port) + ": " + results.message)
                continue
            end if
            if results.len == 0 then
                continue
            end if
            allResults[str(port)] = results
            ui.success("Port " + str(port) + ": " + results.len + " exploit result(s)")

            // Track best shell found so far across all ports
            pick = bestResult(results)
            if pick != null then
                if pick.rtype == "shell" or pick.rtype == "computer" then
                    if privRank(pick.privilege) > bestPriv then
                        bestPriv  = privRank(pick.privilege)
                        bestShell = {}
                        bestShell.port    = port
                        bestShell.result  = pick
                    end if
                end if
            end if
        end for

        return bestShell
    end function

    // ════════════════════════════════════════════════════════════════════
    // ── runRouterAttack: unified router exploit function ─────────────────
    // Replaces routerScan + routerMailAttack.
    // autoMode=true:  run all exploits silently, import everything to DB
    // autoMode=false: collect all results, present interactive menu
    runRouterAttack = function(publicIp, lanIp, autoMode)
        if autoMode == null then
            autoMode = false
        end if
        if lanIp == null then
            lanIp = ""
        end if

        ui.separator()
        if autoMode then
            ui.highlight("ROUTER ATTACK  " + publicIp)
        else
            ui.highlight("ROUTER ATTACK  " + publicIp)
        end if
        ui.separator()

        meta = include_lib("/lib/metaxploit.so")
        if meta == null then
            meta = include_lib(current_path + "/metaxploit.so")
        end if
        if meta == null then
            ui.warning("metaxploit.so not available")
            return
        end if

        crypto = include_lib("/lib/crypto.so")
        if crypto == null then
            crypto = include_lib(current_path + "/crypto.so")
        end if

        dbManager2 = mediator.getExtension("databaseManager")

        ui.info("Connecting to router at " + publicIp + "...")
        netSession = meta.net_use(publicIp)
        if not netSession then
            ui.warning("Could not connect to router at " + publicIp)
            return
        end if

        routerLib = netSession.dump_lib
        if not routerLib then
            ui.warning("Could not get router library")
            return
        end if
        kbLibName    = routerLib.lib_name
        kbLibVersion = routerLib.version
        ui.success("Router library: " + kbLibName + " v" + kbLibVersion)

        // ── Build exploit list from KB or full scan ───────────────────────
        toAttempt = []
        kbRecords = dbManager2.kbLookup(kbLibName, kbLibVersion)
        if kbRecords.len > 0 then
            ui.success("KB hit: " + kbLibName + " v" + kbLibVersion + " - " + str(kbRecords.len) + " known exploit(s)")
            for kr in kbRecords
                if kr.outcome == "number_result" then
                    continue
                end if
                if kr.outcome == "version_mismatch" then
                    continue
                end if
                if kr.outcome == "uncertain" and kr.hasIndex("fail_count") and kr.fail_count > 3 then
                    continue
                end if
                entry = {}
                entry.addr = kr.address
                entry.vn   = kr.exploit
                entry.constraints = []
                if kr.hasIndex("constraints") and kr.constraints.len > 0 then
                    for cs in kr.constraints.split(" ")
                        if cs.len > 0 then
                            entry.constraints.push(cs)
                        end if
                    end for
                end if
                toAttempt.push(entry)
            end for
        else
            // Full scan
            addresses = meta.scan(routerLib)
            if addresses == null or addresses.len == 0 then
                ui.warning("No vulnerable addresses found")
                ui.info("Check hackshop for kernel_router " + kbLibVersion + " exploits")
                return
            end if
            ui.success("Found " + str(addresses.len) + " vulnerable address(es) - scanning...")
            for addr in addresses
                scanText = meta.scan_address(routerLib, addr)
                if scanText == null then
                    continue
                end if
                vulnNames = []
                chunks = scanText.split("Unsafe check: ")
                for chunk in chunks
                    if chunk == chunks[0] then
                        continue
                    end if
                    bStart = chunk.indexOf("<b>")
                    bEnd   = chunk.indexOf("</b>")
                    if bStart != null and bEnd != null then
                        vn = chunk[bStart+3:bEnd]
                        if vn.len > 0 then
                            vulnNames.push(vn)
                        end if
                    end if
                end for
                constraints = []
                for scanLine in scanText.split(char(10))
                    trimmed = scanLine.trim
                    if trimmed.indexOf("* ") == 0 then
                        constraints.push(trimmed[2:])
                    end if
                end for
                for vn in vulnNames
                    entry = {}
                    entry.addr = addr
                    entry.vn   = vn
                    entry.constraints = constraints
                    toAttempt.push(entry)
                end for
            end for
        end if

        if toAttempt.len == 0 then
            ui.warning("No exploits to try")
            return
        end if

        ui.info("Trying " + str(toAttempt.len) + " exploit(s)...")
        ui.separator()

        // ── Try ALL exploits, collect every successful result ─────────────
        collected = []
        for attempt in toAttempt
            addr = attempt.addr
            vn   = attempt.vn

            if lanIp.len > 0 then
                result = routerLib.overflow(addr, vn, lanIp)
            else
                result = routerLib.overflow(addr, vn)
            end if

            resultType = typeof(result)

            if resultType == "number" then
                dbManager2.kbRecord(kbLibName, kbLibVersion, addr, vn, "number_result", attempt.constraints, "number", "", "", "")
                continue
            end if
            if result == null or resultType == "string" then
                dbManager2.kbRecord(kbLibName, kbLibVersion, addr, vn, "uncertain", attempt.constraints, "Failed", "", "", "")
                continue
            end if

            // Record success
            dbManager2.kbRecord(kbLibName, kbLibVersion, addr, vn, "success", attempt.constraints, resultType, "", "guest", "")
            dbManager2.save()

            entry = {}
            entry.addr       = addr
            entry.vn         = vn
            entry.result     = result
            entry.resultType = resultType
            collected.push(entry)

            ui.success("  [" + str(collected.len - 1) + "] " + resultType + " <- " + vn)
        end for

        if collected.len == 0 then
            ui.print("")
            ui.warning("All exploits failed - constraints may not be met")
            ui.info("  Active user on network? Enough computers on network?")
            ui.separator()
            return
        end if

        dbManager2.save()

        // ── AUTO MODE: process everything silently ────────────────────────
        if autoMode then
            harvestedIps = []
            for entry in collected
                // Skip duplicate computer results for same LAN IP
                if entry.resultType == "computer" then
                    if harvestedIps.indexOf(lanIp) != null then
                        ui.dim("  Skipping duplicate computer result for " + lanIp)
                        continue
                    end if
                    harvestedIps.push(lanIp)
                end if
                routerProcessResult(entry.result, entry.resultType, publicIp, lanIp, crypto, dbManager2)
            end for
            // ── Post-result sudo escalation check ─────────────────────────
            // If routerattack yielded a shell session on this IP and we now
            // have a cracked root password in DB (from computer/file harvest
            // above), attempt Strategy 3 sudo escalation automatically.
            routerSudoAttempt(publicIp, lanIp, dbManager2)
            ui.separator()
            return
        end if

        // ── INTERACTIVE MODE: present menu ────────────────────────────────
        ui.print("")
        ui.highlight("RESULTS FOUND")
        ui.separator()
        for i in range(0, collected.len - 1)
            entry = collected[i]
            rType = entry.resultType
            desc  = ""
            if rType == "computer" then
                desc = "extract mail / passwd / bank"
            else if rType == "shell" then
                desc = "guest shell on router - adopt as session"
            else if rType == "file" then
                fPath = ""
                if not (entry.result isa Error) then
                    fPath = " at " + entry.result.path
                end if
                desc = "file/folder access" + fPath + " - usernames / passwd"
            else
                desc = rType
            end if
            ui.info("  [" + str(i) + "] " + rType + "  ->  " + desc)
        end for

        ui.print("")
        ui.info("Options:")
        ui.print("  all    - action all results")
        ui.print("  0-" + str(collected.len - 1) + "    - action specific result")
        ui.print("  skip   - cancel")
        ui.print("")
        choice = user_input("  Choice: ").trim.lower

        if choice == "skip" or choice == "s" then
            return
        end if

        toAction = []
        if choice == "all" then
            toAction = collected[0:]
        else
            idx = choice.val
            if idx >= 0 and idx < collected.len then
                toAction = [collected[idx]]
            else
                ui.warning("Invalid choice")
                return
            end if
        end if

        for entry in toAction
            routerProcessResult(entry.result, entry.resultType, publicIp, lanIp, crypto, dbManager2)
        end for

        // ── Post-result sudo escalation check ─────────────────────────────
        // Same check as autoMode: if we have a shell session on this IP
        // and a cracked root password in DB, attempt Strategy 3 now.
        routerSudoAttempt(publicIp, lanIp, dbManager2)

        ui.separator()
    end function

    // ── routerSudoAttempt: attempt sudo escalation after router harvest ───
    // Called after all router results processed. Checks if we now have both
    // an active shell session on publicIp AND a cracked root password in DB.
    // If so, fires the get_shell GCO trick to escalate without SSH port.
    routerSudoAttempt = function(publicIp, lanIp, dbManager2)
        ui             = mediator.getExtension("corvidUi")
        sessionManager = mediator.getExtension("sessionManager")

        // Find if we have a live non-root shell session on this IP
        targetShell = null
        for sess in sessionManager.sessions
            if sess == sessionManager.pivotSession then
                continue
            end if
            if sess.data.ip.public != publicIp then
                continue
            end if
            if not (sess.handler isa ShellSession) then
                continue
            end if
            sessPriv = sess.handler.privilege()
            if sessPriv == "root" then
                // Already root - nothing to do
                return
            end if
            targetShell = sess.handler.shell.shell
        end for

        if targetShell == null then
            return
        end if

        // Check DB for a cracked root password for this IP
        rootPass3 = ""
        allCreds3 = dbManager2.findCredentialsByIp(publicIp)
        for c3 in allCreds3
            if c3.type == "system" and c3.username == "root" and c3.password.len > 0 then
                rootPass3 = c3.password
            end if
        end for
        if lanIp.len > 0 then
            lanCreds3 = dbManager2.findCredentialsByIp(lanIp)
            for c3 in lanCreds3
                if c3.type == "system" and c3.username == "root" and c3.password.len > 0 then
                    rootPass3 = c3.password
                end if
            end for
        end if

        if rootPass3.len == 0 then
            return
        end if

        ui.info("Router harvest found root password - attempting sudo escalation...")

        wp3r = findWritePerms(targetShell)
        if wp3r == null then
            ui.dim("  Sudo escalation: no writable path on target - skipping")
            return
        end if

        // Embed credentials directly in script source to avoid params
        // space-splitting issues with native passwords
        q3r = char(34)
        sudoLines3 = "cob = get_custom_object" + char(10)
        sudoLines3 = sudoLines3 + "cob.shell = get_shell(" + q3r + "root" + q3r + ", " + q3r + rootPass3 + q3r + ")"

        targetShell.host_computer.touch(wp3r.path, "csudo.src")
        sudoFile3 = targetShell.host_computer.File(wp3r.path + "/csudo.src")
        if sudoFile3 == null then
            ui.dim("  Sudo escalation: could not create script - skipping")
            return
        end if

        sudoFile3.set_content(sudoLines3)
        buildRes3 = targetShell.build(wp3r.path + "/csudo.src", wp3r.path)
        if buildRes3.len > 0 then
            ui.dim("  Sudo escalation: build failed - " + buildRes3)
            deadFile3 = targetShell.host_computer.File(wp3r.path + "/csudo.src")
            if deadFile3 != null then
                deadFile3.delete()
            end if
            return
        end if

        targetShell.launch(wp3r.path + "/csudo")
        cob3r = get_custom_object
        if cob3r.hasIndex("shell") and typeof(cob3r.shell) == "shell" then
            rootShell3 = cob3r.shell
            for s3rname in ["csudo.src", "csudo"]
                s3rf = targetShell.host_computer.File(wp3r.path + "/" + s3rname)
                if s3rf != null then
                    s3rf.delete()
                end if
            end for
            newSess3r = sessionManager.addSession(rootShell3, "root@" + publicIp)
            if not (newSess3r isa Error) then
                sessionManager.setCurrentSession(newSess3r)
                updateData3r = {}
                updateData3r.public_ip = publicIp
                updateData3r.lan_ip    = lanIp
                updateData3r.privilege = "root"
                updateData3r.status    = TARGET_STATUS.EXPLOITED
                dbManager2.upsertTarget(updateData3r)
                credUpdate3r = {}
                credUpdate3r.type      = CRED_TYPES.SYSTEM
                credUpdate3r.username  = "root"
                credUpdate3r.password  = rootPass3
                credUpdate3r.public_ip = publicIp
                credUpdate3r.lan_ip    = lanIp
                credUpdate3r.source    = "sudo_escalation"
                dbManager2.upsertCredential(credUpdate3r)
                dbManager2.save()
                ui.success("Escalated to root via Clover Method after router harvest")
            end if
        else
            ui.dim("  Sudo escalation: get_shell returned no shell")
            for s3rname in ["csudo.src", "csudo"]
                s3rf2 = targetShell.host_computer.File(wp3r.path + "/" + s3rname)
                if s3rf2 != null then
                    s3rf2.delete()
                end if
            end for
        end if
    end function

    // ── routerProcessResult: action a single router exploit result ────────
    routerProcessResult = function(result, resultType, publicIp, lanIp, crypto, dbManager2)
        ui      = mediator.getExtension("corvidUi")

    finalPriv  = sessionManager.currentSession.handler.privilege()
    finalLabel = sessionManager.currentSession.label

    ui.print("")
    ui.info("Public target  : " + publicIp + " (" + entryPriv + ")")
    if hasLan then
        ui.info("LAN target     : " + lanTarget + " (" + lanPriv + ")")
    end if
    ui.info("Current session: " + finalLabel + " as " + finalPriv)
    ui.info("DB totals      : " + str(stats.credentials) + " creds, " + str(stats.persons) + " persons, " + str(stats.targets) + " targets")
    ui.print("")
    ui.sessions(sessionManager.listSessions())
    ui.print("")

    // Prompt to mark as owned - only if root AND SSH is open (useful for bounce network)
    autoOwnCfg = configManager.get("autoOwn")
    if autoOwnCfg isa Error then
        autoOwnCfg = "1"
    end if
    if finalPriv == "root" and autoOwnCfg == "1" then
        finalIpOwn = sessionManager.currentSession.data.ip.public
        // Check if SSH port is open and forwarded before offering owned prompt
        hasSsh = false
        ownCheckTargets = dbManager.findTarget(finalIpOwn)
        for ownTgt in ownCheckTargets
            if ownTgt.hasIndex("ports") then
                for ownPr in ownTgt.ports
                    if ownPr.service == "ssh" and ownPr.forwarded then
                        hasSsh = true
                    end if
                end for
            end if
        end for
        if hasSsh then
            ownAnswer = user_input("  Harvest complete. Mark " + finalIpOwn + " as owned for bounce network? (yes/no): ")
            if ownAnswer.trim == "yes" or ownAnswer.trim == "y" then
                ownTargets = dbManager.findTarget(finalIpOwn)
                if ownTargets.len > 0 then
                    dbManager.setTargetStatus(ownTargets[0].id, "owned")
                else
                    ownData = {}
                    ownData.public_ip = finalIpOwn
                    ownData.lan_ip    = sessionManager.currentSession.data.ip.local
                    ownData.status    = "owned"
                    dbManager.upsertTarget(ownData)
                end if
                dbManager.save()
                ui.success("Marked as owned. Run 'bounces update' to add to bounce network.")
            end if
            ui.print("")
        end if
    end if

    ui.info("Session open. Run harvest, lannmap, escalate, or leave when done.")
    ui.separator()
end function
commandManager.register(_cmd)
