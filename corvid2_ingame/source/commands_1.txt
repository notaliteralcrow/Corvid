// --- activate ---
_cmd = new Command
_cmd.name = "activate"
_cmd.description = "Open an interactive terminal on the current session"
_cmd.parameters = []
_cmd.permissions = {}
_cmd.permissions.object = "shell"
_cmd.permissions.privilege = "guest"
_cmd.callback = function(params, mediator)
        sessionManager = mediator.getExtension("sessionManager")
        ui             = mediator.getExtension("corvidUi")
        dbManager      = mediator.getExtension("databaseManager")
        handler        = sessionManager.currentSession.handler
        if not (handler isa ShellSession) then
            return Error.New("activate requires a shell session", "NOT_SHELL")
        end if
        handler.activate()
        // After returning from native terminal, check session is still alive
        // A reboot during activate will kill the shell - fall back gracefully
        rawShell = handler.shell.shell
        sessionDead = false
        if rawShell == null then
            sessionDead = true
        else
            testComp = rawShell.host_computer
            if testComp == null then
                sessionDead = true
            end if
        end if
        if sessionDead then
            ui.warning("Session lost after activate (machine may have rebooted) - falling back")
            sessionManager.back()
            dbManager.save()
        end if
    end function
commandManager.register(_cmd)

// --- corruptlogs ---
_cmd = new Command
_cmd.name = "corruptlogs"
_cmd.description = "Wipe system.log, trigger a benign regen entry, then open LogViewer to manually clear it."
_cmd.parameters = []
_cmd.permissions = {}
_cmd.permissions.object = "shell"
_cmd.permissions.privilege = "root"
_cmd.callback = function(params, mediator)
        sessionManager = mediator.getExtension("sessionManager")
        ui             = mediator.getExtension("corvidUi")
        handler        = sessionManager.currentSession.handler
        rawShell       = handler.shell.shell
        rawComp        = rawShell.host_computer

        // Step 1: wipe system.log with move trick
        result = handler.corruptLogs()
        if result isa Error then
            return result
        end if
        ui.success("Log wiped.")

        // Step 2: touch and delete a temp file to regenerate system.log
        // with a single benign entry before we open LogViewer
        rawComp.touch("/var", "corvid_tmp")
        tmpFile = rawComp.File("/var/corvid_tmp")
        if tmpFile != null then
            tmpFile.delete()
        end if
        ui.info("Log regenerated with single entry.")

        // Step 3: launch LogViewer so user can manually clear the last entry
        ui.info("Opening LogViewer — delete the remaining entry, then save.")
        logViewerPath = "/usr/bin/LogViewer.exe"
        lvFile = rawComp.File(logViewerPath)
        if lvFile == null then
            logViewerPath = "/bin/LogViewer.exe"
            lvFile = rawComp.File(logViewerPath)
        end if
        if lvFile == null then
            ui.warning("LogViewer.exe not found — clear the remaining log entry manually.")
        else
            rawShell.launch(logViewerPath)
        end if
    end function
commandManager.register(_cmd)

// --- nmap ---
_cmd = new Command
_cmd.name = "nmap"
_cmd.description = "Scan a target IP and display ports"
_cmd.parameters = [StringParameter.New({"name": "ip", "required": true})]
_cmd.permissions = {}
_cmd.permissions.object = "file"
_cmd.permissions.privilege = "guest"
_cmd.callback = function(params, mediator)
        ui        = mediator.getExtension("corvidUi")
        dbManager = mediator.getExtension("databaseManager")
        ip        = params.ip

        if not is_valid_ip(ip) then
            return Error.New("Invalid IP address: " + ip, "INVALID_IP")
        end if

        // Detect private/LAN IP ranges - get_router() always uses Nest context
        // so nmapping a LAN IP from a remote session returns Nest data, not target data
        isPrivate = false
        if ip.indexOf("192.168.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("10.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.16.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.17.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.18.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.19.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.20.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.21.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.22.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.23.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.24.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.25.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.26.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.27.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.28.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.29.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.30.") == 0 then
            isPrivate = true
        end if
        if ip.indexOf("172.31.") == 0 then
            isPrivate = true
        end if

        if isPrivate then
            ui.warning("Cannot nmap LAN IP " + ip + " directly - nmap always runs from your Nest context")
            ui.info("To scan LAN machines, use: attack <public_ip> <lan_ip>")
            ui.info("Or use portattack <public_ip> <port> <lan_ip> from inside the network")
            ui.print("")
            // Show any cached data from DB for this LAN IP
            cached = dbManager.findTarget(ip)
            if cached.len > 0 then
                ui.highlight("CACHED DATA FOR " + ip)
                ui.separator()
                for tgt in cached
                    if tgt.hasIndex("ports") and tgt.ports.len > 0 then
                        ui.info("  Source public IP: " + tgt.public_ip)
                        for pr in tgt.ports
                            fwdStr = "0"
                            if pr.forwarded then
                                fwdStr = "1"
                            end if
                            ui.print("  port " + str(pr.port) + "  " + pr.service + "  " + pr.version + "  forwarded:" + fwdStr)
                        end for
                    end if
                end for
                ui.separator()
            else
                ui.dim("No cached data for " + ip + " in DB")
            end if
            return
        end if

        router = Router.New(ip)
        if router isa Error then
            router = Router.Local()
            if router isa Error then
                return router
            end if
        end if

        ui.info("Router: " + router.publicIp() + " | " + router.localIp() + " | kernel " + router.kernelVersion())
        rules = router.firewallRules()
        if rules.len > 0 then
            ui.warning("Firewall rules: " + rules.join(", "))
        else
            ui.success("No firewall rules")
        end if
        print()

        devices = router.devices()
        if devices isa Error then
            return devices
        end if

        // Display each device and its ports as a table
        for kv in devices
            deviceIp = kv.key
            ports    = kv.value
            ui.info("Device: " + deviceIp)

            rows = []
            keys = ["port", "service", "version", "forwarded", "open"]
            for port in ports
                fwdColor = "#555555"
                if port.forwarded then
                    fwdColor = "#00ff00"
                end if
                openColor = "#555555"
                if port.open then
                    openColor = "#00ff00"
                end if
                row = {}
                row.port = {}
                row.port.text = str(port.port)
                row.port.color = "#FFFC33"
                row.service = {}
                row.service.text = port.service
                row.service.color = "#058FAB"
                row.version = {}
                row.version.text = port.version
                row.version.color = "#555555"
                row.forwarded = {}
                row.forwarded.text = str(port.forwarded)
                row.forwarded.color = fwdColor
                row.open = {}
                row.open.text = str(port.open)
                row.open.color = openColor
                rows.push(row)

                // Auto-feed target database
                targetData = {}
                targetData.public_ip = ip
                targetData.lan_ip    = deviceIp
                targetData.router_ip = router.publicIp()
                portRecord = {}
                portRecord.port      = port.port
                portRecord.service   = port.service
                portRecord.version   = port.version
                portRecord.forwarded = port.forwarded
                portRecord.exploited = false
                targetData.ports = [portRecord]
                // Flag machines with native SSH for bounce revival sweep
                if port.service == "ssh" then
                    targetData.ssh_default = true
                end if
                dbManager.upsertTarget(targetData)
            end for
            ui.table(keys, rows)
        end for
        dbManager.save()
    end function
commandManager.register(_cmd)

// --- fs ---
_cmd = new Command
_cmd.name = "fs"
_cmd.description = "Browse file system of current session with permission indicators. Usage: fs [path]"
_cmd.parameters = [StringParameter.New({"name": "path", "required": false, "default": "/"})]
_cmd.permissions = {}
_cmd.permissions.object = "file"
_cmd.permissions.privilege = "guest"
_cmd.callback = function(params, mediator)
    ui             = mediator.getExtension("corvidUi")
    sessionManager = mediator.getExtension("sessionManager")

    session = sessionManager.currentSession
    if session == null then
        return Error.New("No active session", "NO_SESSION")
    end if

    handler = session.handler
    startPath = params.path.trim
    if startPath.len == 0 then
        startPath = "/"
    end if

    // Get raw computer object for file access - avoids facade get_files/get_folders issue
    // ShellSession gives us shell.shell.host_computer for raw game API
    rawComp = null
    if session.handler isa ShellSession then
        rawComp = session.handler.shell.shell.host_computer
    end if
    if rawComp == null then
        return Error.New("fs requires a shell session", "NOT_SHELL")
    end if

    // Get starting file via raw computer File() method
    startFile = rawComp.File(startPath)
    if startFile == null then
        return Error.New("Cannot access path: " + startPath, "PATH_ERROR")
    end if

    ui.separator()
    ui.highlight("FILE SYSTEM  [" + startPath + "]")
    ui.separator()

    // Recursive tree printer - uses raw game file objects (.name .path as properties)
    // maxDepth kept low to avoid hitting GreyScript execution time limit
    maxDepth = 2
    fsPrint = function(f, depth, prefix)
        if depth > maxDepth then
            return
        end if
        if f == null then
            return
        end if

        // Raw game file objects: has_permission is a method
        readable = f.has_permission("r")
        writable = f.has_permission("w")
        execable = f.has_permission("x")

        r = " " ; w = " " ; x = " "
        if readable then
            r = "r"
        end if
        if writable then
            w = "w"
        end if
        if execable then
            x = "x"
        end if
        permLabel = "[" + r + w + x + "]"

        // .name and .is_folder are properties on raw game file objects
        fname = f.name
        if f.is_folder then
            fname = fname + "/"
        end if

        sizeStr = ""
        if not f.is_folder then
            sizeStr = "  " + f.size
        end if

        if readable and writable then
            ui.success(prefix + fname + "  " + permLabel + sizeStr)
        else if readable then
            ui.info(prefix + fname + "  " + permLabel + sizeStr)
        else
            ui.dim(prefix + fname + "  " + permLabel + sizeStr)
        end if

        // Only recurse into readable folders
        if f.is_folder and readable then
            childPrefix = prefix + "  "
            subFolders = f.get_folders
            subFiles   = f.get_files
            if subFolders != null and subFolders.len > 0 then
                for sf in subFolders
                    fsPrint(sf, depth + 1, childPrefix)
                end for
            end if
            if subFiles != null and subFiles.len > 0 then
                for fl in subFiles
                    fsPrint(fl, depth + 1, childPrefix)
                end for
            end if
        end if
    end function

    fsPrint(startFile, 0, "  ")
    ui.print("")
    ui.dim("  [rwx] = read/write/execute  |  green=writable  cyan=readable  dim=no access")
    ui.dim("  Depth: " + str(maxDepth) + "  |  Use: fs /lib  fs /home  fs /etc for specific paths")
    ui.separator()
end function
commandManager.register(_cmd)

// --- decipher ---
// Crack a hash string directly without needing a session.
// Accepts either a raw hash, a full "username:hash" passwd line,
// or a file path - reads the file and deciphers each line.
// Usage: decipher <hash>  or  decipher <username:hash>  or  decipher <filepath>
_cmd = new Command
_cmd.name = "decipher"
_cmd.description = "Crack a password hash. Accepts hash, username:hash, or a file path to decipher all lines."
_cmd.parameters = [StringParameter.New({"name": "input", "required": true, "default": ""})]
_cmd.permissions = {}
_cmd.permissions.object = "file"
_cmd.permissions.privilege = "guest"
_cmd.callback = function(params, mediator)
        ui             = mediator.getExtension("corvidUi")
        intelManager   = mediator.getExtension("intelManager")
        sessionManager = mediator.getExtension("sessionManager")

        input = params.input.trim
        if input.len == 0 then
            return Error.New("Provide a hash, username:hash, or file path", "MISSING_PARAM")
        end if

        // Detect file path - starts with / or ./ and exists on current session
        lines = []
        if input.indexOf("/") == 0 or input.indexOf("./") == 0 then
            handler = sessionManager.currentSession.handler
            f = handler.getFile(input)
            if not (f isa Error) then
                content = f.content()
                if not (content isa Error) and content.len > 0 then
                    for line in content.split(char(10))
                        line = line.trim
                        if line.len > 0 then
                            lines.push(line)
                        end if
                    end for
                end if
            end if
        end if

        // If not a file or file empty, treat input as a single hash
        if lines.len == 0 then
            lines.push(input)
        end if

        ui.separator()
        cracked = 0
        failed  = 0
        for line in lines
            username = ""
            hash = line
            colonIdx = line.indexOf(":")
            atIdx = line.indexOf("@")
            // "user@domain.com:hash" - email format, split on last colon
            // "username:hash" - passwd format
            if colonIdx != null then
                if atIdx != null and atIdx < colonIdx then
                    // email:hash format
                    username = line[:colonIdx]
                    hash = line[colonIdx + 1:]
                else if atIdx == null then
                    // username:hash format
                    username = line[:colonIdx]
                    hash = line[colonIdx + 1:]
                end if
            end if

            if hash.len == 0 then
                continue
            end if

            if username.len > 0 then
                ui.info("Deciphering: " + username)
            else
                ui.info("Deciphering hash: " + hash[:20] + "...")
            end if

            result = intelManager.crackHash(username, hash)
            if result isa Error then
                ui.warning("  Could not crack")
                failed = failed + 1
            else
                if username.len > 0 then
                    ui.success("  " + username + " : " + result)
                else
                    ui.success("  Password: " + result)
                end if
                cracked = cracked + 1
            end if
        end for

        if lines.len > 1 then
            ui.print("")
            ui.info("Cracked: " + str(cracked) + "  Failed: " + str(failed))
        end if
        ui.separator()
    end function
commandManager.register(_cmd)

    ui.info("Lib/versions covered: " + str(stats.libVersions))
    ui.print("")

    // Show per-lib breakdown
    allRecords = dbManager.getExploitKb().db.all()
    // Group by lib_name + lib_version
    groups = {}
    for r in allRecords
        if not r.hasIndex("lib_name") then
            continue
        end if
        if filterLib.len > 0 and r.lib_name.indexOf(filterLib) == null then
            continue
        end if
        key = r.lib_name + " v" + r.lib_version
        if not groups.hasIndex(key) then
            groups[key] = {"success": 0, "blocked": 0, "uncertain": 0, "total": 0}
        end if
        groups[key].total = groups[key].total + 1
        o = "uncertain"
        if r.hasIndex("outcome") then
            o = r.outcome
        end if
        if o == "success" then
            groups[key].success = groups[key].success + 1
        else if o == "constraint_blocked" then
            groups[key].blocked = groups[key].blocked + 1
        else
            groups[key].uncertain = groups[key].uncertain + 1
        end if
    end for

    if groups.indexes.len == 0 then
        if filterLib.len > 0 then
            ui.warning("No KB records matching: " + filterLib)
        else
            ui.warning("No exploit KB records yet - run attacks to populate")
        end if
        return
    end if

    ui.highlight("LIB/VERSION BREAKDOWN")
    ui.separator()
    for key in groups.indexes
        g = groups[key]
        statusStr = ""
        if g.success > 0 then
            statusStr = statusStr + str(g.success) + " success  "
        end if
        if g.blocked > 0 then
            statusStr = statusStr + str(g.blocked) + " blocked  "
        end if
        if g.uncertain > 0 then
            statusStr = statusStr + str(g.uncertain) + " uncertain"
        end if
        if g.success > 0 then
            ui.success("  " + key + "  [" + statusStr.trim + "]")
        else if g.blocked > 0 then
            ui.warning("  " + key + "  [" + statusStr.trim + "]")
        else
            ui.dim("  " + key + "  [" + statusStr.trim + "]")
        end if
    end for
    ui.separator()
end function
commandManager.register(_cmd)

// --- dbmigrate ---
_cmd = new Command
_cmd.name = "dbmigrate"
_cmd.description = "Migrate existing credential records to add native_pass and source tags. Safe to run multiple times."
_cmd.parameters = []
_cmd.permissions = {}
_cmd.permissions.object = "file"
_cmd.permissions.privilege = "guest"
_cmd.callback = function(params, mediator)
    ui        = mediator.getExtension("corvidUi")
    dbManager = mediator.getExtension("databaseManager")

    ui.separator()
    ui.highlight("DATABASE MIGRATION")
    ui.separator()
    ui.info("Scanning credential records for migration...")

    allCreds = dbManager.allCredentials()
    updated  = 0
    skipped  = 0

    for rec in allCreds
        needsUpdate = false

        // Add native_pass field if missing
        if not rec.hasIndex("native_pass") then
            rec.native_pass = ""
            needsUpdate = true
        end if

        // Tag source field if missing or unknown
        if not rec.hasIndex("source") or rec.source.len == 0 or rec.source == "unknown" then
            // Assume existing untagged records are native passwords
            // (Corvid-generated passwords would have been tagged going forward)
            rec.source = "native"
            needsUpdate = true
        end if

        // If source is "harvest" (old tag) convert to "native"
        if rec.hasIndex("source") and rec.source == "harvest" then
            rec.source = "native"
            needsUpdate = true
        end if

        // If source is "bounce" (old tag) convert to "hardened"
        if rec.hasIndex("source") and rec.source == "bounce" then
            rec.source = "hardened"
            needsUpdate = true
        end if

        // If source is one of the old router scan tags, convert to "native"
        if rec.hasIndex("source") then
            if rec.source == "router_scan" or rec.source == "router_lib_exploit" or rec.source == "router_lib_home" or rec.source == "router_home_scan" or rec.source == "portattack_lib" or rec.source == "portattack_home" then
                rec.source = "native"
                needsUpdate = true
            end if
        end if

        if needsUpdate then
            dbManager.updateCredentialRecord(rec)
            updated = updated + 1
        else
            skipped = skipped + 1
        end if
    end for

    dbManager.save()
    ui.print("")
    ui.success("Migration complete.")
    ui.info("  Updated : " + str(updated) + " records")
    ui.info("  Skipped : " + str(skipped) + " (already migrated)")
    ui.separator()
end function
commandManager.register(_cmd)

// --- dbclean ---
_cmd = new Command
_cmd.name = "dbclean"
_cmd.description = "Deduplicate bloated notes fields across all target records"
_cmd.parameters = []
_cmd.permissions = {}
_cmd.permissions.object = "file"
_cmd.permissions.privilege = "guest"
_cmd.callback = function(params, mediator)
    ui        = mediator.getExtension("corvidUi")
    dbManager = mediator.getExtension("databaseManager")

    ui.separator()
    ui.highlight("DATABASE CLEAN")
    ui.separator()

    targets = dbManager.allTargets()
    cleaned = 0
    totalSaved = 0

    for target in targets
        if not target.hasIndex("notes") then
            continue
        end if
        notes = target.notes
        if notes == null or notes.len == 0 then
            continue
        end if

        // Deduplicate notes - process line by line
        lines = notes.split("
")
        seenLines = []
        cleanLines = []

        for line in lines
            line = line.trim
            if line.len == 0 then
                continue
            end if

            // For bounce: lines - strip duplicate source: and health: tokens
            if line.indexOf("bounce:") == 0 then
                parts = line.split(" ")
                cleanParts = []
                seenSource = false
                seenHealth = false
                for part in parts
                    if part.indexOf("source:") == 0 then
                        if not seenSource then
                            cleanParts.push(part)
                            seenSource = true
                        end if
                    else if part.indexOf("health:") == 0 then
                        if not seenHealth then
                            cleanParts.push(part)
                            seenHealth = true
                        end if
                    else
                        cleanParts.push(part)
                    end if
                end for
                line = cleanParts.join(" ")
            end if

            // Skip duplicate whole lines
            alreadySeen = false
            for seen in seenLines
                if seen == line then
                    alreadySeen = true
                end if
            end for
            if not alreadySeen then
                cleanLines.push(line)
                seenLines.push(line)
            end if
        end for

        newNotes = cleanLines.join("
")
        if newNotes != notes then
            saved = notes.len - newNotes.len
            totalSaved = totalSaved + saved
            target.notes = newNotes
            dbManager.targetDb.db.set(target.id, target)
            dbManager.targetDb.db.dirty = true
            cleaned = cleaned + 1
            ui.info("  Cleaned " + target.id + " (" + str(saved) + " chars removed)")
        end if
    end for

    if cleaned > 0 then
        dbManager.save()
        ui.success("Cleaned " + str(cleaned) + " records")
        ui.success("Removed " + str(totalSaved) + " characters of duplicate data")
    else
        ui.info("No bloated records found - DB is clean")
    end if
    ui.separator()
end function
commandManager.register(_cmd)
