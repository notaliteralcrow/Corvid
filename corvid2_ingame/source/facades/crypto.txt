// ============================================================
// CORVID 2.0 | facades/crypto
// Wraps the game cryptoLib object.
// decipher() takes BOTH username and hash as required
// by the API  -  crypto.decipher(username, hash).
// ============================================================

Crypto = {}
Crypto.crypto = null

Crypto.decipher = function(username, hash)
    result = self.crypto.decipher(hash)
    if result == null then
        return Error.New("Could not decipher hash for: " + username, "DECIPHER_FAILED")
    end if
    return result
end function

Crypto.aircrack = function(filepath)
    result = self.crypto.aircrack(filepath)
    if result == null then
        return Error.New("aircrack failed on: " + filepath, "AIRCRACK_ERROR")
    end if
    return result
end function

Crypto.aireplay = function(bssid, essid, maxAcks)
    return self.crypto.aireplay(bssid, essid, maxAcks)
end function

Crypto.airmon = function(option, iface)
    return self.crypto.airmon(option, iface)
end function

Crypto.smtpUserList = function(ip, port)
    result = self.crypto.smtp_user_list(ip, port)
    if result == null then
        return Error.New("smtp_user_list failed", "SMTP_ERROR")
    end if
    return result
end function

Crypto.New = function(object)
    if typeof(object) != "cryptoLib" then
        return Error.New("Cannot create Crypto from non-cryptoLib object", "TYPE_ERROR")
    end if
    result = new Crypto
    result.crypto = object
    return result
end function
