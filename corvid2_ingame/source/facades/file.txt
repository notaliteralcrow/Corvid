// ============================================================
// CORVID 2.0 | facades/file
// Wraps the game File object.
// All methods return typed results  -  never raw null or strings.
// Directories are Files with type FOLDER.
// ============================================================

FILE_TYPES = {}
FILE_TYPES.FILE   = "FILE"
FILE_TYPES.FOLDER = "FOLDER"

File = {}
File.file = null

File.name = function()
    return self.file.name
end function

File.path = function()
    return self.file.path
end function

File.owner = function()
    return self.file.owner
end function

File.group = function()
    return self.file.group
end function

File.permissions = function()
    return self.file.permissions
end function

File.size = function()
    return self.file.size
end function

File.type = function()
    if self.file.is_folder then
        return FILE_TYPES.FOLDER
    end if
    return FILE_TYPES.FILE
end function

File.isBinary = function()
    return self.file.is_binary == 1
end function

File.hasPermission = function(permission)
    return self.file.has_permission(permission)
end function

File.content = function()
    if self.type() == FILE_TYPES.FOLDER then
        return Error.New("Cannot read content of a folder", "TYPE_ERROR")
    end if
    if self.isBinary() then
        return Error.New("File is binary and cannot be read as text", "BINARY_FILE")
    end if
    result = self.file.get_content
    if result == null then
        return Error.New("Could not read file content", "READ_ERROR")
    end if
    return result
end function

File.setContent = function(newContent)
    if self.type() == FILE_TYPES.FOLDER then
        return Error.New("Cannot set content of a folder", "TYPE_ERROR")
    end if
    result = self.file.set_content(newContent)
    if not result then
        return Error.New("Could not write file content", "WRITE_ERROR")
    end if
end function

File.appendContent = function(newContent)
    if self.type() == FILE_TYPES.FOLDER then
        return Error.New("Cannot append content to a folder", "TYPE_ERROR")
    end if
    existing = self.content()
    if existing isa Error then
        return existing
    end if
    return self.setContent(existing + newContent)
end function

File.children = function()
    if self.type() == FILE_TYPES.FILE then
        return Error.New("Cannot get children of a file", "TYPE_ERROR")
    end if
    raw = self.file.get_files() + self.file.get_folders()
    wrapChild = function(child)
        return File.New(child)
    end function
    return map(raw, @wrapChild)
end function

File.child = function(name)
    kids = self.children()
    if kids isa Error then
        return kids
    end if
    matchName = function(child)
        return child.name() == name
    end function
    result = find(kids, @matchName)
    if result == null then
        return Error.New("Child not found: " + name, "NOT_FOUND")
    end if
    return result
end function

File.hasChild = function(name)
    result = self.child(name)
    return not (result isa Error)
end function

File.parent = function()
    raw = self.file.parent
    if raw == null then
        return Error.New("No parent  -  already at root", "AT_ROOT")
    end if
    return File.New(raw)
end function

File.copy = function(destination, name)
    if name == null then
        name = self.name()
    end if
    result = self.file.copy(destination, name)
    if typeof(result) == "string" then
        return Error.New(result, "COPY_ERROR")
    end if
end function

File.move = function(destination, name)
    if name == null then
        name = self.name()
    end if
    result = self.file.move(destination, name)
    if typeof(result) == "string" then
        return Error.New(result, "MOVE_ERROR")
    end if
end function

File.rename = function(newName)
    result = self.file.rename(newName)
    if typeof(result) == "string" and result.len > 0 then
        return Error.New(result, "RENAME_ERROR")
    end if
end function

File.delete = function()
    if self.type() == FILE_TYPES.FOLDER then
        kids = self.children()
        if not (kids isa Error) then
            for child in kids
                child.delete()
            end for
        end if
    end if
    if self.path() == "/" then
        return
    end if
    result = self.file.delete()
    if typeof(result) == "string" and result.len > 0 then
        return Error.New(result, "DELETE_ERROR")
    end if
end function

File.setPermissions = function(perms, recursive)
    if recursive == null then
        recursive = false
    end if
    result = self.file.chmod(perms, recursive)
    if typeof(result) == "string" and result.len > 0 then
        return Error.New(result, "CHMOD_ERROR")
    end if
end function

File.setOwner = function(owner, recursive)
    if recursive == null then
        recursive = false
    end if
    result = self.file.set_owner(owner, recursive)
    if typeof(result) == "string" and result.len > 0 then
        return Error.New(result, "CHOWN_ERROR")
    end if
end function

File.setGroup = function(group, recursive)
    if recursive == null then
        recursive = false
    end if
    result = self.file.set_group(group, recursive)
    if typeof(result) == "string" and result.len > 0 then
        return Error.New(result, "CHGRP_ERROR")
    end if
end function

File.New = function(file)
    if typeof(file) != "file" then
        return Error.New("Cannot create File from non-file object", "TYPE_ERROR")
    end if
    result = new File
    result.file = file
    return result
end function
