// ============================================================
// CORVID 2.0 | facades/router
// Wraps the game Router object.
// devices() returns a fully annotated map of all devices
// and their ports including service type, version,
// forwarding status, and open/closed state.
// ============================================================

Router = {}
Router.router = null

Router.publicIp = function()
    return self.router.public_ip
end function

Router.localIp = function()
    return self.router.local_ip
end function

Router.kernelVersion = function()
    return self.router.kernel_version
end function

Router.bssid = function()
    return self.router.bssid_name
end function

Router.essid = function()
    return self.router.essid_name
end function

Router.firewallRules = function()
    return self.router.firewall_rules
end function

Router.usedPorts = function()
    return self.router.used_ports
end function

Router.deviceIps = function()
    return self.router.devices_lan_ip
end function

Router.devicePorts = function(lanIp)
    return self.router.device_ports(lanIp)
end function

Router.portInfo = function(port)
    raw = self.router.port_info(port)
    result = {}
    parts = raw.split(" ")
    if parts.len >= 1 then
        result.service = parts[0]
    end if
    if parts.len >= 2 then
        result.version = parts[1]
    end if
    if result.service == null then
        result.service = "unknown"
    end if
    if result.version == null then
        result.version = "unknown"
    end if
    return result
end function

Router.pingPort = function(portNumber)
    return self.router.ping_port(portNumber)
end function

Router.devices = function()
    ips        = self.deviceIps()
    usedPorts  = self.usedPorts()
    router     = self.router

    isForwarded = function(port)
        checkPort = function(usedPort)
            return usedPort.get_lan_ip == port.get_lan_ip and usedPort.port_number == port.port_number
        end function
        return some(usedPorts, @checkPort)
    end function

    buildPortInfo = function(port)
        info    = router.port_info(port)
        parts   = info.split(" ")
        result  = {}
        result.port      = port.port_number
        result.lanIp     = port.get_lan_ip
        result.open      = not port.is_closed
        result.forwarded = isForwarded(port)
        result.service   = "unknown"
        result.version   = "unknown"
        if parts.len >= 1 then
            result.service = parts[0]
        end if
        if parts.len >= 2 then
            result.version = parts[1]
        end if
        return result
    end function

    devices = {}
    for ip in ips
        rawPorts = self.devicePorts(ip)
        portList = []
        for rawPort in rawPorts
            portList.push(buildPortInfo(rawPort))
        end for
        devices[ip] = portList
    end for

    // Add the router itself as a device on port 0
    routerIp = self.localIp()
    if not devices.hasIndex(routerIp) then
        devices[routerIp] = []
    end if
    routerEntry = {}
    routerEntry.port      = 0
    routerEntry.lanIp     = routerIp
    routerEntry.open      = true
    routerEntry.forwarded = true
    routerEntry.service   = "kernel_router"
    routerEntry.version   = self.kernelVersion()
    devices[routerIp].push(routerEntry)

    return devices
end function

Router.New = function(ip)
    if not is_valid_ip(ip) then
        return Error.New("Invalid IP address: " + ip, "INVALID_IP")
    end if
    // Check if current shell has internet before calling get_router
    s = get_shell
    if s == null then
        return Error.New("No active shell - cannot reach router for: " + ip, "NO_SHELL")
    end if
    raw = get_router(ip)
    if raw == null then
        return Error.New("No router found for: " + ip + " (check internet connection)", "NOT_FOUND")
    end if
    result = new Router
    result.router = raw
    return result
end function

Router.Local = function()
    s = get_shell
    if s == null then
        return Error.New("No active shell - cannot reach local router", "NO_SHELL")
    end if
    raw = get_router()
    if raw == null then
        return Error.New("Could not find local router (check internet connection)", "NOT_FOUND")
    end if
    result = new Router
    result.router = raw
    return result
end function
