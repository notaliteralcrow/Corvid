// ============================================================
// CORVID 2.0 | facades/metaxploit
// Wraps the game MetaxploitLib object.
// hack() provides a fully automated scan-and-exploit chain.
// getVulnerabilities() returns structured exploit data.
// ============================================================

Metaxploit = {}
Metaxploit.metaxploit = null

Metaxploit.netUse = function(ip, port)
    if port == null then
        session = self.metaxploit.net_use(ip)
    else
        session = self.metaxploit.net_use(ip, port)
    end if
    if session == null then
        return Error.New("Could not connect to " + ip, "CONNECT_ERROR")
    end if
    return session
end function

Metaxploit.load = function(path)
    result = self.metaxploit.load(path)
    if result == null then
        return Error.New("Could not load library from: " + path, "LOAD_ERROR")
    end if
    return result
end function

Metaxploit.scan = function(metaLib)
    result = self.metaxploit.scan(metaLib)
    if result == null then
        return Error.New("Scan returned null  -  library may be unvulnerable", "SCAN_ERROR")
    end if
    return result
end function

Metaxploit.scanAddress = function(metaLib, address)
    result = self.metaxploit.scan_address(metaLib, address)
    if result == null then
        return Error.New("scan_address returned null for: " + address, "SCAN_ERROR")
    end if
    return result
end function

Metaxploit.flockClient = function(ip, port, processName)
    if processName == null then
        processName = "ps"
    end if
    result = self.metaxploit.rshell_client(ip, port, processName)
    if result == false then
        return Error.New("Failed to launch flock client", "FLOCK_ERROR")
    end if
end function

Metaxploit.flockServer = function()
    result = self.metaxploit.rshell_server()
    if typeof(result) == "string" then
        return Error.New(result, "FLOCK_ERROR")
    end if
    return result
end function

Metaxploit.sniffer = function(saveEncSource)
    return self.metaxploit.sniffer(saveEncSource)
end function

Metaxploit.getVulnerabilities = function(metaLib)
    // Returns a list of {address, exploits[]} objects
    addresses = self.scan(metaLib)
    if addresses isa Error then
        return addresses
    end if
    if addresses.len == 0 then
        return Error.New("No vulnerable regions found", "NO_VULNS")
    end if

    results = []
    for address in addresses
        scanResult = self.scanAddress(metaLib, address)
        if scanResult isa Error then
            continue
        end if

        exploits  = []
        segments  = scanResult.split("Unsafe check: ")[1:]
        for segment in segments
            labelStart = segment.indexOf("<b>")
            labelEnd   = segment.indexOf("</b>")
            if labelStart == null or labelEnd == null then
                continue
            end if
            value = segment[labelStart + 3 : labelEnd]
            if value.len > 0 then
                exploits.push(value)
            end if
        end for

        if exploits.len > 0 then
            entry = {}
            entry.address = address
            entry.exploits = exploits
            results.push(entry)
        end if
    end for

    if results.len == 0 then
        return Error.New("No parseable exploit values found", "NO_EXPLOITS")
    end if
    return results
end function

Metaxploit.hack = function(metaLib)
    // Full automated exploit chain.
    // Returns a list of wrapped result objects sorted by quality.
    vulns = self.getVulnerabilities(metaLib)
    if vulns isa Error then
        return vulns
    end if

    results = []
    for vuln in vulns
        for exploit in vuln.exploits
            result = metaLib.overflow(vuln.address, exploit)
            if result == null or typeof(result) == "number" or typeof(result) == "string" then
                continue
            end if
            entry = {}
            entry.address = vuln.address
            entry.exploit = exploit
            entry.raw     = result
            entry.rtype   = typeof(result)
            results.push(entry)
        end for
    end for

    if results.len == 0 then
        return Error.New("All exploits failed", "EXPLOIT_FAILED")
    end if
    return results
end function

Metaxploit.New = function(object)
    if typeof(object) != "MetaxploitLib" then
        return Error.New("Cannot create Metaxploit from non-MetaxploitLib object", "TYPE_ERROR")
    end if
    result = new Metaxploit
    result.metaxploit = object
    return result
end function
