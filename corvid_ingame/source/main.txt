init = function()
    // 1  -  Add Nest as first session (pivot)
    nestShell = get_shell
    session   = sessionManager.addSession(nestShell, "Nest")
    if session isa Error then
        exit("CORVID: Could not establish Nest session: " + session.message)
    end if

    // 2  -  Load config first (theme depends on it)
    configResult = configManager.init()
    if configResult isa Error then
        print("[!] Config init failed: " + configResult.message + "  -  using defaults")
    end if

    // 3  -  Init theme from config
    themeManager.init()

    // 4  -  Init database (needs pivot session)
    dbResult = databaseManager.init()
    if dbResult isa Error then
        exit("CORVID: Database init failed: " + dbResult.message)
    end if

    // 5  -  Load metaxploit.so from Nest
    nestHandler = sessionManager.pivotSession.handler
    metaFile    = nestHandler.getFile("/lib/metaxploit.so")
    if metaFile isa Error then
        metaFile = nestHandler.getFile("metaxploit.so")
    end if
    if not (metaFile isa Error) then
        rawMeta = include_lib(metaFile.path())
        if rawMeta != null then
            ipData = {}
            ipData.ip = sessionManager.pivotSession.data.ip
            ipData.path = metaFile.path()
            libManager.addLib(rawMeta, ipData)
        end if
    end if

    // 6  -  Load crypto.so from Nest
    cryptoFile = nestHandler.getFile("/lib/crypto.so")
    if cryptoFile isa Error then
        cryptoFile = nestHandler.getFile("crypto.so")
    end if
    if not (cryptoFile isa Error) then
        rawCrypto = include_lib(cryptoFile.path())
        if rawCrypto != null then
            ipData = {}
            ipData.ip = sessionManager.pivotSession.data.ip
            ipData.path = cryptoFile.path()
            libManager.addLib(rawCrypto, ipData)
        end if
    end if

    // 7  -  Load saved aliases from config
    aliases = configManager.get("aliases")
    if not (aliases isa Error) then
        for kv in aliases
            commandManager.addAlias(kv.key, kv.value)
        end for
    end if

    // 8  -  Display banner and status
    clear_screen
    corvidUi.init()

    // 9  -  Init roost manager
    roostManager.init()

    // 10 -  Init shinylibs manager
    shinyLibsManager.init(mediator)

    // 10a - Baseline shinylibs scan of Nest libs if DB is empty
    autoShiny = configManager.get("autoShiny")
    if autoShiny isa Error then
        autoShiny = "1"
        configManager.set("autoShiny", "1")
    end if

    // Default apt repo for flock client auto-install of metaxploit
    flockAptRepo = configManager.get("flockAptRepo")
    if flockAptRepo isa Error then
        configManager.set("flockAptRepo", "30.225.197.154")
    end if
    if autoShiny == "1" and shinyLibsManager.db.records.len == 0 then
        nestHandler = sessionManager.pivotSession.handler
        nestComp    = nestHandler.shell.shell.host_computer
        nestLibDir  = nestComp.File("/lib")
        if nestLibDir != null and nestLibDir.is_folder then
            corvidUi.dim("Scanning Nest /lib/ for shinylibs baseline...")
            libFiles = nestLibDir.get_files
            if libFiles != null then
                rawMeta = include_lib("/lib/metaxploit.so")
                if rawMeta != null then
                    for nestLib in libFiles
                        if nestLib.name.indexOf(".so") == null then
                            continue
                        end if
                        metalib = rawMeta.load(nestLib.path)
                        if metalib == null then
                            continue
                        end if
                        addrs = rawMeta.scan(metalib)
                        if addrs == null then
                            addrs = []
                        end if
                        exploitTexts = []
                        for addr in addrs
                            txt = rawMeta.scan_address(metalib, addr)
                            if txt != null then
                                exploitTexts.push(txt)
                            end if
                        end for
                        shinyLibsManager.db.stage(nestLib.name, metalib.version, exploitTexts, nestLib.path)
                    end for
                end if
            end if
            // Auto-collect all staged Nest libs into DB (no SCP needed - already local)
            for staged in shinyLibsManager.db.staged
                shinyLibsManager.db.collect(staged.libName, staged.version, staged.srcPath)
            end for
            databaseManager.save()
            corvidUi.dim("Shinylibs baseline: " + str(shinyLibsManager.db.records.len) + " lib(s) recorded from Nest.")
        end if
    end if

    // 11 -  Run init macro if present
    macroManager.init()
end function

            rawShell = sessHandler.shell.shell
            if rawShell == null then
                sessionDead = true
            else
                testComp = rawShell.host_computer
                if testComp == null then
                    sessionDead = true
                end if
            end if
        end if
        if sessionDead then
            corvidUi.warning("Session lost (machine rebooted?) - falling back to previous session")
            sessionManager.back()
            databaseManager.save()
            showMainMenu = true
            continue
        end if
    end if

                else if sAnswer == "4" then
                    cur = configManager.get("autoOwn")
                    if cur isa Error or cur == "1" then
                        configManager.set("autoOwn", "0")
                        corvidUi.warning("autoOwn OFF — no owned prompt at end of attack")
                    else
                        configManager.set("autoOwn", "1")
                        corvidUi.success("autoOwn ON")
                    end if
                else if sAnswer == "5" then
                    cur = configManager.get("kbFallback")
                    if cur isa Error or cur == "1" then
                        configManager.set("kbFallback", "0")
                        corvidUi.warning("kbFallback OFF — only KB-known exploits will be tried")
                    else
                        configManager.set("kbFallback", "1")
                        corvidUi.success("kbFallback ON — full scan runs when no KB hits")
                    end if
                else if sAnswer == "6" then
                    corvidUi.separator()
                    corvidUi.highlight("ALIASES")
                    corvidUi.separator()
                    commandManager.execute("aliases")
                    commandManager.run()
                else if sAnswer == "7" then
                    corvidUi.info("Usage: theme <slot> <hex>")
                    corvidUi.info("Slots: highlight  text  secondary  success  warning  info  danger  error")
                    commandManager.run()
                end if
            end while

        end if

        // After any menu action, return to main menu
        showMainMenu = true

    else
        // -- Direct command --------------------------------------------------------
        commandManager.execute(answer)
        commandManager.run()

        // Post-command session check
        currentSess2 = sessionManager.currentSession
        if currentSess2 != null and currentSess2 != sessionManager.pivotSession then
            sessPriv = currentSess2.handler.privilege()
            if sessPriv == null or sessPriv == "" then
                corvidUi.warning("Session connection lost - falling back to previous session")
                sessionManager.back()
                databaseManager.save()
            end if
        end if

    end if
end while
