// ============================================================
// CORVID 2.0 | managers/sniffer/sniffer-db
// Persistent store for sniffer captures.
// Separate from credential DB - captures are staged here
// until promoted to main creds via 'sniffer promote'.
// ============================================================

SnifferDb = {}
SnifferDb.records  = []
SnifferDb.mediator = null
SnifferDb.nextId   = 1

SnifferDb.parse = function(raw)
    // Parse meta.sniffer(0) output into structured record
    // Format:
    //   Incoming data captured.
    //   Source: <ip>
    //   Destination: <ip>
    //   Protocol: <port>
    //   USER: <user>
    //   PASSWORD: <pass>
    result = {}
    result.source_ip  = ""
    result.dest_ip    = ""
    result.protocol   = ""
    result.username   = ""
    result.password   = ""

    lines = raw.split(char(10))
    for line in lines
        line = line.trim
        if line.indexOf("Source: ") == 0 then
            result.source_ip = line[8:]
        else if line.indexOf("Destination: ") == 0 then
            result.dest_ip = line[13:]
        else if line.indexOf("Protocol: ") == 0 then
            result.protocol = line[10:]
        else if line.indexOf("USER: ") == 0 then
            result.username = line[6:]
        else if line.indexOf("PASSWORD: ") == 0 then
            result.password = line[10:]
        end if
    end for

    return result
end function

SnifferDb.add = function(raw, capturedFrom)
    parsed = self.parse(raw)
    record = {}
    record.id           = "sn_" + str(self.nextId)
    record.source_ip    = parsed.source_ip
    record.dest_ip      = parsed.dest_ip
    record.protocol     = parsed.protocol
    record.username     = parsed.username
    record.password     = parsed.password
    record.captured_from = capturedFrom
    record.timestamp    = str(floor(time))
    record.raw          = raw
    self.records.push(record)
    self.nextId = self.nextId + 1
    return record
end function

SnifferDb.find = function(id)
    for r in self.records
        if r.id == id then
            return r
        end if
    end for
    return null
end function

SnifferDb.remove = function(id)
    newRecords = []
    for r in self.records
        if r.id != id then
            newRecords.push(r)
        end if
    end for
    self.records = newRecords
end function

SnifferDb.load = function(data)
    if data.hasIndex("sniffer_captures") then
        self.records = data.sniffer_captures
        if self.records.len > 0 then
            lastId = self.records[self.records.len - 1].id
            numStr = lastId[3:]
            self.nextId = numStr.val + 1
        end if
    end if
end function

SnifferDb.save = function(data)
    data.sniffer_captures = self.records
end function

SnifferDb.setMediator = function(mediator)
    self.mediator = mediator
end function
