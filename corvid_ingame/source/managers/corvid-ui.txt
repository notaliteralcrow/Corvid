// ---- Prompt ----

CorvidUi.prompt = function(session)
    theme           = self.mediator.getExtension("themeManager")
    platformManager = self.mediator.getExtension("platformManager")
    handler = session.handler
    ip      = session.data.ip
    name    = ""

    if handler.type() == "shell" or handler.type() == "computer" then
        name = handler.name()
    end if

    user    = handler.user
    pubIp   = ip.public
    lanIp   = ip.local
    path    = handler.path()
    hType   = handler.type()
    priv    = handler.privilege()

    promptStr = theme.secondary("(") + theme.highlight(user) + theme.secondary("@") + theme.highlight(name) + theme.secondary(")") + theme.secondary("[") + theme.info(pubIp) + theme.secondary("|") + theme.info(lanIp) + theme.secondary("]") + theme.secondary("[") + theme.warning(path) + theme.secondary("]") + theme.secondary("[") + theme.success(priv) + theme.secondary("]")

    // Show platform mode indicator when active
    if platformManager != null and platformManager.active then
        promptStr = promptStr + theme.secondary("[") + theme.warning("via ") + theme.highlight(platformManager.nodeIp) + theme.secondary("]")
    end if

    promptStr = promptStr + theme.secondary(" ") + theme.highlight("$ ")

    return promptStr
end function

// ---- Table rendering ----

CorvidUi.table = function(keys, rows)
    theme = self.mediator.getExtension("themeManager")

    // Calculate column widths
    widths = {}
    for key in keys
        widths[key] = key.len
    end for
    for row in rows
        for key in keys
            if row.hasIndex(key) then
                cellLen = (row[key].text + "").len
                if cellLen > widths[key] then
                    widths[key] = cellLen
                end if
            end if
        end for
    end for

    // Header
    headerStr = ""
    for key in keys
        headerStr = headerStr + theme.highlight(padRight(upper(key), widths[key] + 2))
    end for
    print(headerStr)

    // Separator
    sepStr = ""
    for key in keys
        dashes = ""
        for i in range(0, key.len - 1)
            dashes = dashes + "-"
        end for
        sepStr = sepStr + theme.secondary(padRight(dashes, widths[key] + 2))
    end for
    print(sepStr)

    // Rows
    for row in rows
        rowStr = ""
        for key in keys
            cell = {"text": "", "color": null}
            if row.hasIndex(key) then
                cell = row[key]
            end if
            padded = padRight(cell.text + "", widths[key] + 2)
            if cell.color != null then
                rowStr = rowStr + "<color=" + cell.color + ">" + padded + "</color>"
            else
                rowStr = rowStr + theme.text(padded)
            end if
        end for
        print(rowStr)
    end for
    print()
end function

// ---- File system tree ----

CorvidUi.fileSystemTree = function(file, depth)
    theme = self.mediator.getExtension("themeManager")
    if depth == null then
        depth = 0
    end if

    indent = ""
    for i in range(0, depth - 1)
        indent = indent + "  "
    end for
    if depth > 0 then
        indent = indent + "-"
    end if

    nameColor = theme.colors.info
    if file.type() == FILE_TYPES.FILE then
        nameColor = theme.colors.secondary
    end if

    line = indent + " " + theme.warning(file.permissions()) + " " + theme.success(file.owner()) + " " + "<color=" + nameColor + ">" + file.name() + "</color>" + " " + theme.secondary(str(file.size()))
    print(line)

    if file.type() == FILE_TYPES.FOLDER then
        kids = file.children()
        if not (kids isa Error) then
            for kid in kids
                self.fileSystemTree(kid, depth + 1)
            end for
        end if
    end if
end function

CorvidUi.credentials = function(records)
    theme = self.mediator.getExtension("themeManager")
    keys  = ["id", "type", "username", "password", "native", "tag", "ip", "lan"]
    rows  = []
    for r in records
        // Current password display
        passDisplay = r.password
        if passDisplay.len == 0 then
            if r.hasIndex("hash") and r.hash.len > 0 then
                hashPreview = r.hash
                if hashPreview.len > 8 then
                    hashPreview = hashPreview[:8]
                end if
                passDisplay = "[" + hashPreview + "...]"
            else
                passDisplay = "(none)"
            end if
        end if
        passColor = theme.colors.success
        if r.password.len == 0 then
            passColor = theme.colors.secondary
        end if

        // Native password display
        nativeDisplay = ""
        nativeColor   = theme.colors.secondary
        if r.hasIndex("native_pass") and r.native_pass.len > 0 then
            nativeDisplay = r.native_pass
            nativeColor   = theme.colors.warning
        end if

        // Source tag and colour
        src = "native"
        if r.hasIndex("source") then
            src = r.source
        end if
        tagColor = theme.colors.secondary
        tagText  = src
        if src == "native"            then
            tagColor = theme.colors.success  ; tagText = "[native]"
        end if
        if src == "exploit_generated" then
            tagColor = theme.colors.error    ; tagText = "[exploit]"
        end if
        if src == "hardened"          then
            tagColor = theme.colors.warning  ; tagText = "[hardened]"
        end if
        if src == "harvest_updated"   then
            tagColor = theme.colors.warning  ; tagText = "[updated]"
        end if
        if src == "sniffer"           then
            tagColor = theme.colors.info     ; tagText = "[sniffed]"
        end if
        if src == "mail"              then
            tagColor = theme.colors.info     ; tagText = "[mail]"
        end if

        row = {}
        row.id = {}
        row.id.text = r.id
        row.id.color = theme.colors.warning
        row.type = {}
        row.type.text = r.type
        row.type.color = theme.colors.info
        row.username = {}
        row.username.text = r.username
        row.username.color = theme.colors.highlight
        row.password = {}
        row.password.text = passDisplay
        row.password.color = passColor
        row.native = {}
        row.native.text = nativeDisplay
        row.native.color = nativeColor
        row.tag = {}
        row.tag.text = tagText
        row.tag.color = tagColor
        row.ip = {}
        row.ip.text = r.public_ip
        row.ip.color = theme.colors.secondary
        row.lan = {}
        lanDisplay = ""
        if r.hasIndex("lan_ip") then
            lanDisplay = r.lan_ip
        end if
        row.lan.text = lanDisplay
        row.lan.color = theme.colors.secondary
        rows.push(row)
    end for
    self.table(keys, rows)
end function

CorvidUi.help = function(commands)
    theme = self.mediator.getExtension("themeManager")
    keys  = ["command", "args", "description"]
    rows  = []
    for cmd in commands
        paramStr = ""
        for param in cmd.parameters
            if param.type == "flag" then
                paramStr = paramStr + "(" + param.flag + ") "
            else if param.required then
                paramStr = paramStr + "[" + param.name + "] "
            else
                paramStr = paramStr + "(" + param.name + ") "
            end if
        end for
        nameColor = theme.colors.error
        if cmd.hasPermissions then
            nameColor = theme.colors.success
        end if
        row = {}
        cmdName = cmd.name
        row.command = {}
        row.command.text = cmdName
        row.command.color = nameColor
        trimmedArgs = paramStr.trim
        row.args = {}
        row.args.text = trimmedArgs
        row.args.color = theme.colors.info
        row.description = {}
        row.description.text = cmd.description
        row.description.color = theme.colors.secondary
        rows.push(row)
    end for
    self.table(keys, rows)
end function

CorvidUi.init = function()
    self.showBanner()
    libManager = self.mediator.getExtension("libManager")
    dbManager  = self.mediator.getExtension("databaseManager")
    theme      = self.mediator.getExtension("themeManager")

    // Library status
    meta = libManager.getCurrentLib(LIB_TYPES.METAXPLOIT)
    if meta != null then
        print(theme.success("[+] metaxploit.so loaded"))
    else
        print(theme.warning("[!] metaxploit.so not found"))
    end if

    crypto = libManager.getCurrentLib(LIB_TYPES.CRYPTO)
    if crypto != null then
        print(theme.success("[+] crypto.so loaded"))
    else
        print(theme.warning("[!] crypto.so not found"))
    end if

    // DB stats
    print()
    stats = dbManager.stats()
    self.dbStats(stats)
    print()
end function

CorvidUi.showSettingsMenu = function(configManager)
    theme = self.mediator.getExtension("themeManager")
    self.separator()
    print(theme.highlight("  CORVID SETTINGS"))
    self.separator()

    // Read current values
    autoShiny  = configManager.get("autoShiny")
    autoWhois  = configManager.get("autoWhois")
    autoSmtp   = configManager.get("autoSmtp")
    autoOwn    = configManager.get("autoOwn")
    kbFallback = configManager.get("kbFallback")
    if autoShiny isa Error then
        autoShiny = "1"
    end if
    if autoWhois isa Error then
        autoWhois = "1"
    end if
    if autoSmtp isa Error then
        autoSmtp = "1"
    end if
    if autoOwn isa Error then
        autoOwn = "1"
    end if
    if kbFallback isa Error then
        kbFallback = "1"
    end if

    onStr  = theme.success("ON ")
    offStr = theme.warning("OFF")

    self.separator()
    print(theme.highlight("  AUTOMATION TOGGLES"))
    self.separator()

    s1 = onStr
    if autoShiny != "1" then
        s1 = offStr
    end if
    s2 = onStr
    if autoWhois != "1" then
        s2 = offStr
    end if
    s3 = onStr
    if autoSmtp != "1" then
        s3 = offStr
    end if
    s4 = onStr
    if autoOwn != "1" then
        s4 = offStr
    end if
    s5 = onStr
    if kbFallback != "1" then
        s5 = offStr
    end if

    print("  [1]  " + s1 + "  " + theme.text("autoShiny  ") + theme.secondary("score libs during port scan + baseline Nest scan on first run"))
    print("  [2]  " + s2 + "  " + theme.text("autoWhois  ") + theme.secondary("whois lookup during guided attack Phase 1"))
    print("  [3]  " + s3 + "  " + theme.text("autoSmtp   ") + theme.secondary("SMTP user enumeration when smtp port detected during attack"))
    print("  [4]  " + s4 + "  " + theme.text("autoOwn    ") + theme.secondary("prompt to mark target as owned at end of attack"))
    print("  [5]  " + s5 + "  " + theme.text("kbFallback ") + theme.secondary("run full port scan when no KB hits found"))
    self.separator()
    print(theme.highlight("  OTHER"))
    self.separator()
    print("  [6]  " + theme.text("Aliases     ") + theme.secondary("view and manage command shortcuts"))
    print("  [7]  " + theme.text("Theme       ") + theme.secondary("customise colours  (theme <slot> <hex>)"))
    self.separator()
    print(theme.secondary("  Select a toggle to flip it, 6-7 for other settings, 0 to go back."))
    print()
end function
