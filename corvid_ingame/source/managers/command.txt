// ============================================================
// PARAMETER TYPES
// ============================================================

PARAMETER_TYPES = {}
PARAMETER_TYPES.STRING  = "string"
PARAMETER_TYPES.NUMBER  = "number"
PARAMETER_TYPES.BOOLEAN = "boolean"
PARAMETER_TYPES.REST    = "rest"
PARAMETER_TYPES.FLAG    = "flag"
PARAMETER_TYPES.OPTIONS = "options"

// ---- StringParameter ----

StringParameter = {}
StringParameter.name     = null
StringParameter.required = false
StringParameter.default  = null
StringParameter.type     = PARAMETER_TYPES.STRING
StringParameter.handle = function(args, index)
    if not args.hasIndex(index) then
        if self.required then
            return Error.New("Missing required parameter: " + self.name, "MISSING_PARAM")
        end if
        return [self.default, index + 1]
    end if
    return [args[index], index + 1]
end function
StringParameter.New = function(options)
    p = new StringParameter
    if not options.hasIndex("name") then
        return Error.New("Parameter requires a name", "MISSING_PARAM")
    end if
    p.name     = options.name
    p.required = options.hasIndex("required") and options.required
    p.default  = null
    if options.hasIndex("default") then
        p.default = options.default
    end if
    return p
end function

// ---- NumberParameter ----

NumberParameter = {}
NumberParameter.name     = null
NumberParameter.required = false
NumberParameter.default  = null
NumberParameter.type     = PARAMETER_TYPES.NUMBER
NumberParameter.handle = function(args, index)
    if not args.hasIndex(index) then
        if self.required then
            return Error.New("Missing required parameter: " + self.name, "MISSING_PARAM")
        end if
        return [self.default, index + 1]
    end if
    raw = args[index]
    num = raw.val
    if typeof(num) == "string" then
        return Error.New("Parameter '" + self.name + "' must be a number", "TYPE_ERROR")
    end if
    return [num, index + 1]
end function
NumberParameter.New = function(options)
    p = new NumberParameter
    if not options.hasIndex("name") then
        return Error.New("Parameter requires a name", "MISSING_PARAM")
    end if
    p.name     = options.name
    p.required = options.hasIndex("required") and options.required
    p.default  = null
    if options.hasIndex("default") then
        p.default = options.default
    end if
    return p
end function

// ---- BooleanParameter ----

BooleanParameter = {}
BooleanParameter.name     = null
BooleanParameter.required = false
BooleanParameter.default  = null
BooleanParameter.type     = PARAMETER_TYPES.BOOLEAN
BooleanParameter.handle = function(args, index)
    if not args.hasIndex(index) then
        if self.required then
            return Error.New("Missing required parameter: " + self.name, "MISSING_PARAM")
        end if
        return [self.default, index + 1]
    end if
    raw = args[index]
    if raw == "true"  then
        return [true,  index + 1]
    end if
    if raw == "false" then
        return [false, index + 1]
    end if
    return Error.New("Parameter '" + self.name + "' must be true or false", "TYPE_ERROR")
end function
BooleanParameter.New = function(options)
    p = new BooleanParameter
    if not options.hasIndex("name") then
        return Error.New("Parameter requires a name", "MISSING_PARAM")
    end if
    p.name     = options.name
    p.required = options.hasIndex("required") and options.required
    p.default  = null
    if options.hasIndex("default") then
        p.default = options.default
    end if
    return p
end function

// ---- FlagParameter ----
// A boolean switch  -  e.g. -R, -f, --verbose
// Returns true if the flag token is present, false otherwise.
// Does NOT advance the index if the flag is absent.

FlagParameter = {}
FlagParameter.name     = null
FlagParameter.required = false
FlagParameter.flag     = null
FlagParameter.type     = PARAMETER_TYPES.FLAG
FlagParameter.handle = function(args, index)
    if not args.hasIndex(index) then
        return [false, index]
    end if
    if args[index] == self.flag then
        return [true, index + 1]
    end if
    return [false, index]
end function
FlagParameter.New = function(options)
    p = new FlagParameter
    if not options.hasIndex("name") then
        return Error.New("Parameter requires a name", "MISSING_PARAM")
    end if
    if not options.hasIndex("flag") then
        return Error.New("FlagParameter requires a flag value", "MISSING_PARAM")
    end if
    p.name = options.name
    p.flag = options.flag
    return p
end function

// ---- OptionsParameter ----
// An enum  -  only accepts values from a predefined set.

OptionsParameter = {}
OptionsParameter.name     = null
OptionsParameter.required = false
OptionsParameter.default  = null
OptionsParameter.options  = {}
OptionsParameter.type     = PARAMETER_TYPES.OPTIONS
OptionsParameter.handle = function(args, index)
    if not args.hasIndex(index) then
        if self.required then
            return Error.New("Missing required parameter: " + self.name, "MISSING_PARAM")
        end if
        return [self.default, index + 1]
    end if
    arg = args[index]
    if not self.options.hasIndex(arg) then
        return Error.New("Invalid option '" + arg + "' for " + self.name, "INVALID_OPTION")
    end if
    return [self.options[arg], index + 1]
end function
OptionsParameter.New = function(options)
    p = new OptionsParameter
    if not options.hasIndex("name") then
        return Error.New("Parameter requires a name", "MISSING_PARAM")
    end if
    if not options.hasIndex("options") then
        return Error.New("OptionsParameter requires an options map", "MISSING_PARAM")
    end if
    p.name     = options.name
    p.required = options.hasIndex("required") and options.required
    p.options  = options.options
    p.default  = null
    if options.hasIndex("default") then
        p.default = options.default
    end if
    return p
end function

// ---- RestParameter ----
// Captures all remaining arguments as a single joined string.
// Always comes last in a parameter list.

RestParameter = {}
RestParameter.name     = null
RestParameter.required = false
RestParameter.default  = null
RestParameter.type     = PARAMETER_TYPES.REST
RestParameter.handle = function(args, index)
    if not args.hasIndex(index) then
        if self.required then
            return Error.New("Missing required parameter: " + self.name, "MISSING_PARAM")
        end if
        return [self.default, args.len]
    end if
    rest = args[index:]
    return [rest.join(" "), args.len]
end function
RestParameter.New = function(options)
    p = new RestParameter
    if not options.hasIndex("name") then
        return Error.New("Parameter requires a name", "MISSING_PARAM")
    end if
    p.name     = options.name
    p.required = options.hasIndex("required") and options.required
    p.default  = null
    if options.hasIndex("default") then
        p.default = options.default
    end if
    return p
end function

// ============================================================
// COMMAND
// ============================================================

Command = {}
Command.name        = null
Command.description = null
Command.parameters  = []
Command.callback    = null
Command.permissions = {"object": "file", "privilege": "guest"}

Command.execute = function(args, mediator)
    parsed = self.parseArgs(args)
    if parsed isa Error then
        return parsed
    end if
    return self.callback(parsed, mediator)
end function

Command.parseArgs = function(args)
    params = {}
    tokens = args.split(" ")

    // Strip empty tokens
    clean = []
    for t in tokens
        if t.len > 0 then
            clean.push(t)
        end if
    end for
    tokens = clean

    index = 0
    for param in self.parameters
        result = param.handle(tokens, index)
        if result isa Error then
            return result
        end if
        params[param.name] = result[0]
        index = result[1]
    end for
    return params
end function

Command.help = function()
    line = self.name + "  "
    for param in self.parameters
        if param.type == PARAMETER_TYPES.FLAG then
            line = line + "(" + param.flag + ") "
        else if param.required then
            line = line + "[" + param.name + "] "
        else
            line = line + "(" + param.name + ") "
        end if
    end for
    line = line + " -  " + self.description
    return line
end function

Command.New = function(options)
    if not options.hasIndex("name") then
        return Error.New("Command requires a name", "MISSING_PARAM")
    end if
    if not options.hasIndex("callback") then
        return Error.New("Command requires a callback", "MISSING_PARAM")
    end if
    c = new Command
    c.name        = options.name
    c.description = ""
    c.parameters  = []
    c.permissions = {"object": "file", "privilege": "guest"}
    if options.hasIndex("description") then
        c.description = options.description
    end if
    if options.hasIndex("parameters")  then
        c.parameters  = options.parameters
    end if
    if options.hasIndex("permissions") then
        c.permissions = {"object": "file", "privilege": "guest"} + options.permissions
    end if
    c.callback = @options.callback
    return c
end function

Command.Build = function(name, description, params, objPerm, privPerm, callback)
    c = new Command
    c.name        = name
    c.description = description
    c.parameters  = params
    c.permissions = {}
    c.permissions.object    = objPerm
    c.permissions.privilege = privPerm
    c.callback = @callback
    return c
end function
